1//! ChiselRunner
2//!
3//! This module contains the `ChiselRunner` struct, which assists with deploying
4//! and calling the REPL contract on a in-memory REVM instance.
5
6use alloy_primitives::{Address, Bytes, Log, U256};
7use eyre::Result;
8use foundry_evm::{
9    executors::{DeployResult, Executor, RawCallResult},
10    traces::{TraceKind, Traces},
11};
12use revm::interpreter::{return_ok, InstructionResult};
13use std::collections::HashMap;
14
15/// The function selector of the REPL contract's entrypoint, the `run()` function.
16static RUN_SELECTOR: [u8; 4] = [0xc0, 0x40, 0x62, 0x26];
17
18/// The Chisel Runner
19///
20/// Based off of foundry's forge cli runner for scripting.
21/// See: [runner](cli::cmd::forge::script::runner.rs)
22#[derive(Debug)]
23pub struct ChiselRunner {
24    /// The Executor
25    pub executor: Executor,
26    /// An initial balance
27    pub initial_balance: U256,
28    /// The sender
29    pub sender: Address,
30    /// Input calldata appended to `RUN_SELECTOR`
31    pub input: Option<Vec<u8>>,
32}
33
34/// Represents the result of a Chisel REPL run
35#[derive(Debug, Default)]
36pub struct ChiselResult {
37    /// Was the run a success?
38    pub success: bool,
39    /// Transaction logs
40    pub logs: Vec<Log>,
41    /// Call traces
42    pub traces: Traces,
43    /// Amount of gas used in the transaction
44    pub gas_used: u64,
45    /// Map of addresses to their labels
46    pub labeled_addresses: HashMap<Address, String>,
47    /// Return data
48    pub returned: Bytes,
49    /// Called address
50    pub address: Option<Address>,
51    /// EVM State at the final instruction of the `run()` function
52    pub state: Option<(Vec<U256>, Vec<u8>, InstructionResult)>,
53}
54
55/// ChiselRunner implementation
56impl ChiselRunner {
57    /// Create a new [ChiselRunner]
58    ///
59    /// ### Takes
60    ///
61    /// An [Executor], the initial balance of the sender, and the sender's [Address].
62    ///
63    /// ### Returns
64    ///
65    /// A new [ChiselRunner]
66    pub fn new(
67        executor: Executor,
68        initial_balance: U256,
69        sender: Address,
70        input: Option<Vec<u8>>,
71    ) -> Self {
72        Self { executor, initial_balance, sender, input }
73    }
74
75    /// Run a contract as a REPL session
76    ///
77    /// ### Takes
78    ///
79    /// The creation bytecode of the REPL contract
80    ///
81    /// ### Returns
82    ///
83    /// Optionally, a tuple containing the deployed address of the bytecode as well as a
84    /// [ChiselResult] containing information about the result of the call to the deployed REPL
85    /// contract.
86    pub fn run(&mut self, bytecode: Bytes) -> Result<(Address, ChiselResult)> {
87        // Set the sender's balance to [U256::MAX] for deployment of the REPL contract.
88        self.executor.set_balance(self.sender, U256::MAX)?;
89
90        // Deploy an instance of the REPL contract
91        // We don't care about deployment traces / logs here
92        let DeployResult { address, .. } = self
93            .executor
94            .deploy(self.sender, bytecode, U256::ZERO, None)
95            .map_err(|err| eyre::eyre!("Failed to deploy REPL contract:\n{}", err))?;
96
97        // Reset the sender's balance to the initial balance for calls.
98        self.executor.set_balance(self.sender, self.initial_balance)?;
99
100        // Append the input to the `RUN_SELECTOR` to form the calldata
101        let mut calldata = RUN_SELECTOR.to_vec();
102        if let Some(mut input) = self.input.clone() {
103            calldata.append(&mut input);
104        }
105
106        // Call the "run()" function of the REPL contract
107        let call_res = self.call(self.sender, address, Bytes::from(calldata), U256::from(0), true);
108
109        call_res.map(|res| (address, res))
110    }
111
112    /// Executes the call.
113    ///
114    /// This will commit the changes if `commit` is true.
115    ///
116    /// This will return _estimated_ gas instead of the precise gas the call would consume, so it
117    /// can be used as `gas_limit`.
118    ///
119    /// Taken from Forge's script runner.
120    fn call(
121        &mut self,
122        from: Address,
123        to: Address,
124        calldata: Bytes,
125        value: U256,
126        commit: bool,
127    ) -> eyre::Result<ChiselResult> {
128        let fs_commit_changed =
129            if let Some(cheatcodes) = &mut self.executor.inspector_mut().cheatcodes {
130                let original_fs_commit = cheatcodes.fs_commit;
131                cheatcodes.fs_commit = false;
132                original_fs_commit != cheatcodes.fs_commit
133            } else {
134                false
135            };
136
137        let mut res = self.executor.call_raw(from, to, calldata.clone(), value)?;
138        let mut gas_used = res.gas_used;
139        if matches!(res.exit_reason, return_ok!()) {
140            // store the current gas limit and reset it later
141            let init_gas_limit = self.executor.env().tx.gas_limit;
142
143            // the executor will return the _exact_ gas value this transaction consumed, setting
144            // this value as gas limit will result in `OutOfGas` so to come up with a
145            // better estimate we search over a possible range we pick a higher gas
146            // limit 3x of a succeeded call should be safe
147            let mut highest_gas_limit = gas_used * 3;
148            let mut lowest_gas_limit = gas_used;
149            let mut last_highest_gas_limit = highest_gas_limit;
150            while (highest_gas_limit - lowest_gas_limit) > 1 {
151                let mid_gas_limit = (highest_gas_limit + lowest_gas_limit) / 2;
152                self.executor.env_mut().tx.gas_limit = mid_gas_limit;
153                let res = self.executor.call_raw(from, to, calldata.clone(), value)?;
154                match res.exit_reason {
155                    InstructionResult::Revert |
156                    InstructionResult::OutOfGas |
157                    InstructionResult::OutOfFunds => {
158                        lowest_gas_limit = mid_gas_limit;
159                    }
160                    _ => {
161                        highest_gas_limit = mid_gas_limit;
162                        // if last two successful estimations only vary by 10%, we consider this to
163                        // sufficiently accurate
164                        const ACCURACY: u64 = 10;
165                        if (last_highest_gas_limit - highest_gas_limit) * ACCURACY /
166                            last_highest_gas_limit <
167                            1
168                        {
169                            // update the gas
170                            gas_used = highest_gas_limit;
171                            break;
172                        }
173                        last_highest_gas_limit = highest_gas_limit;
174                    }
175                }
176            }
177            // reset gas limit in the
178            self.executor.env_mut().tx.gas_limit = init_gas_limit;
179        }
180
181        // if we changed `fs_commit` during gas limit search, re-execute the call with original
182        // value
183        if fs_commit_changed {
184            if let Some(cheatcodes) = &mut self.executor.inspector_mut().cheatcodes {
185                cheatcodes.fs_commit = !cheatcodes.fs_commit;
186            }
187
188            res = self.executor.call_raw(from, to, calldata.clone(), value)?;
189        }
190
191        if commit {
192            // if explicitly requested we can now commit the call
193            res = self.executor.transact_raw(from, to, calldata, value)?;
194        }
195
196        let RawCallResult { result, reverted, logs, traces, labels, chisel_state, .. } = res;
197
198        Ok(ChiselResult {
199            returned: result,
200            success: !reverted,
201            gas_used,
202            logs,
203            traces: traces
204                .map(|traces| {
205                    // Manually adjust gas for the trace to add back the stipend/real used gas
206
207                    vec![(TraceKind::Execution, traces)]
208                })
209                .unwrap_or_default(),
210            labeled_addresses: labels,
211            address: None,
212            state: chisel_state,
213        })
214    }
215}
