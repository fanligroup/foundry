59/// EVM executor.
60///
61/// The executor can be configured with various `revm::Inspector`s, like `Cheatcodes`.
62///
63/// There are multiple ways of interacting the EVM:
64/// - `call`: executes a transaction, but does not persist any state changes; similar to `eth_call`,
65///   where the EVM state is unchanged after the call.
66/// - `transact`: executes a transaction and persists the state changes
67/// - `deploy`: a special case of `transact`, specialized for persisting the state of a contract
68///   deployment
69/// - `setup`: a special case of `transact`, used to set up the environment for a test
70#[derive(Clone, Debug)]
71pub struct Executor {
72    /// The underlying `revm::Database` that contains the EVM storage.
73    // Note: We do not store an EVM here, since we are really
74    // only interested in the database. REVM's `EVM` is a thin
75    // wrapper around spawning a new EVM on every call anyway,
76    // so the performance difference should be negligible.
77    backend: Backend,
78    /// The EVM environment.
79    env: EnvWithHandlerCfg,
80    /// The Revm inspector stack.
81    inspector: InspectorStack,
82    /// The gas limit for calls and deployments. This is different from the gas limit imposed by
83    /// the passed in environment, as those limits are used by the EVM for certain opcodes like
84    /// `gaslimit`.
85    gas_limit: u64,
86    /// Whether `failed()` should be called on the test contract to determine if the test failed.
87    legacy_assertions: bool,
88}

162    /// Creates the default CREATE2 Contract Deployer for local tests and scripts.
163    pub fn deploy_create2_deployer(&mut self) -> eyre::Result<()> {
164        trace!("deploying local create2 deployer");
165        let create2_deployer_account = self
166            .backend()
167            .basic_ref(DEFAULT_CREATE2_DEPLOYER)?
168            .ok_or_else(|| BackendError::MissingAccount(DEFAULT_CREATE2_DEPLOYER))?;
169
170        // If the deployer is not currently deployed, deploy the default one.
171        if create2_deployer_account.code.map_or(true, |code| code.is_empty()) {
172            let creator = DEFAULT_CREATE2_DEPLOYER_DEPLOYER;
173
174            // Probably 0, but just in case.
175            let initial_balance = self.get_balance(creator)?;
176            self.set_balance(creator, U256::MAX)?;
177
178            let res =
179                self.deploy(creator, DEFAULT_CREATE2_DEPLOYER_CODE.into(), U256::ZERO, None)?;
180            trace!(create2=?res.address, "deployed local create2 deployer");
181
182            self.set_balance(creator, initial_balance)?;
183        }
184        Ok(())
185    }

252    /// Deploys a contract using the given `env` and commits the new state to the underlying
253    /// database.
254    ///
255    /// # Panics
256    ///
257    /// Panics if `env.tx.transact_to` is not `TxKind::Create(_)`.
258    #[instrument(name = "deploy", level = "debug", skip_all)]
259    pub fn deploy_with_env(
260        &mut self,
261        env: EnvWithHandlerCfg,
262        rd: Option<&RevertDecoder>,
263    ) -> Result<DeployResult, EvmError> {
264        assert!(
265            matches!(env.tx.transact_to, TxKind::Create),
266            "Expected create transaction, got {:?}",
267            env.tx.transact_to
268        );
269        trace!(sender=%env.tx.caller, "deploying contract");
270
271        let mut result = self.transact_with_env(env)?;
272        result = result.into_result(rd)?;
273        let Some(Output::Create(_, Some(address))) = result.out else {
274            panic!("Deployment succeeded, but no address was returned: {result:#?}");
275        };
276
277        // also mark this library as persistent, this will ensure that the state of the library is
278        // persistent across fork swaps in forking mode
279        self.backend_mut().add_persistent_account(address);
280
281        debug!(%address, "deployed contract");
282
283        Ok(DeployResult { raw: result, address })
284    }

286    /// Calls the `setUp()` function on a contract.
287    ///
288    /// This will commit any state changes to the underlying database.
289    ///
290    /// Ayn changes made during the setup call to env's block environment are persistent, for
291    /// example `vm.chainId()` will change the `block.chainId` for all subsequent test calls.
292    #[instrument(name = "setup", level = "debug", skip_all)]
293    pub fn setup(
294        &mut self,
295        from: Option<Address>,
296        to: Address,
297        rd: Option<&RevertDecoder>,
298    ) -> Result<RawCallResult, EvmError> {
299        trace!(?from, ?to, "setting up contract");
300
301        let from = from.unwrap_or(CALLER);
302        self.backend_mut().set_test_contract(to).set_caller(from);
303        let calldata = Bytes::from_static(&ITest::setUpCall::SELECTOR);
304        let mut res = self.transact_raw(from, to, calldata, U256::ZERO)?;
305        res = res.into_result(rd)?;
306
307        // record any changes made to the block's environment during setup
308        self.env_mut().block = res.env.block.clone();
309        // and also the chainid, which can be set manually
310        self.env_mut().cfg.chain_id = res.env.cfg.chain_id;
311
312        let success =
313            self.is_raw_call_success(to, Cow::Borrowed(&res.state_changeset), &res, false);
314        if !success {
315            return Err(res.into_execution_error("execution error".to_string()).into());
316        }
317
318        Ok(res)
319    }

401    /// Execute the transaction configured in `env.tx`.
402    #[instrument(name = "transact", level = "debug", skip_all)]
403    pub fn transact_with_env(&mut self, mut env: EnvWithHandlerCfg) -> eyre::Result<RawCallResult> {
404        let mut inspector = self.inspector().clone();
405        let backend = self.backend_mut();
406        let result = backend.inspect(&mut env, &mut inspector)?;
407        let mut result =
408            convert_executed_result(env, inspector, result, backend.has_state_snapshot_failure())?;
409        self.commit(&mut result);
410        Ok(result)
411    }

413    /// Commit the changeset to the database and adjust `self.inspector_config` values according to
414    /// the executed call result.
415    ///
416    /// This should not be exposed to the user, as it should be called only by `transact*`.
417    #[instrument(name = "commit", level = "debug", skip_all)]
418    fn commit(&mut self, result: &mut RawCallResult) {
419        // Persist changes to db.
420        self.backend_mut().commit(result.state_changeset.clone());
421
422        // Persist cheatcode state.
423        self.inspector_mut().cheatcodes = result.cheatcodes.take();
424        if let Some(cheats) = self.inspector_mut().cheatcodes.as_mut() {
425            // Clear broadcastable transactions
426            cheats.broadcastable_transactions.clear();
427            cheats.ignored_traces.ignored.clear();
428
429            // if tracing was paused but never unpaused, we should begin next frame with tracing
430            // still paused
431            if let Some(last_pause_call) = cheats.ignored_traces.last_pause_call.as_mut() {
432                *last_pause_call = (0, 0);
433            }
434        }
435
436        // Persist the changed environment.
437        self.inspector_mut().set_env(&result.env);
438    }

458    /// Returns `true` if a test can be considered successful.
459    ///
460    /// This is the same as [`Self::is_success`], but intended for outcomes of [`Self::call_raw`].
461    pub fn is_raw_call_success(
462        &self,
463        address: Address,
464        state_changeset: Cow<'_, StateChangeset>,
465        call_result: &RawCallResult,
466        should_fail: bool,
467    ) -> bool {
468        if call_result.has_state_snapshot_failure {
469            // a failure occurred in a reverted snapshot, which is considered a failed test
470            return should_fail;
471        }
472        self.is_success(address, call_result.reverted, state_changeset, should_fail)
473    }

475    /// Returns `true` if a test can be considered successful.
476    ///
477    /// If the call succeeded, we also have to check the global and local failure flags.
478    ///
479    /// These are set by the test contract itself when an assertion fails, using the internal `fail`
480    /// function. The global flag is located in [`CHEATCODE_ADDRESS`] at slot [`GLOBAL_FAIL_SLOT`],
481    /// and the local flag is located in the test contract at an unspecified slot.
482    ///
483    /// This behavior is inherited from Dapptools, where initially only a public
484    /// `failed` variable was used to track test failures, and later, a global failure flag was
485    /// introduced to track failures across multiple contracts in
486    /// [ds-test#30](https://github.com/dapphub/ds-test/pull/30).
487    ///
488    /// The assumption is that the test runner calls `failed` on the test contract to determine if
489    /// it failed. However, we want to avoid this as much as possible, as it is relatively
490    /// expensive to set up an EVM call just for checking a single boolean flag.
491    ///
492    /// See:
493    /// - Newer DSTest: <https://github.com/dapphub/ds-test/blob/e282159d5170298eb2455a6c05280ab5a73a4ef0/src/test.sol#L47-L63>
494    /// - Older DSTest: <https://github.com/dapphub/ds-test/blob/9ca4ecd48862b40d7b0197b600713f64d337af12/src/test.sol#L38-L49>
495    /// - forge-std: <https://github.com/foundry-rs/forge-std/blob/19891e6a0b5474b9ea6827ddb90bb9388f7acfc0/src/StdAssertions.sol#L38-L44>
496    pub fn is_success(
497        &self,
498        address: Address,
499        reverted: bool,
500        state_changeset: Cow<'_, StateChangeset>,
501        should_fail: bool,
502    ) -> bool {
503        let success = self.is_success_raw(address, reverted, state_changeset);
504        should_fail ^ success
505    }
506
507    #[instrument(name = "is_success", level = "debug", skip_all)]
508    fn is_success_raw(
509        &self,
510        address: Address,
511        reverted: bool,
512        state_changeset: Cow<'_, StateChangeset>,
513    ) -> bool {
514        // The call reverted.
515        if reverted {
516            return false;
517        }
518
519        // A failure occurred in a reverted snapshot, which is considered a failed test.
520        if self.backend().has_state_snapshot_failure() {
521            return false;
522        }
523
524        // Check the global failure slot.
525        if let Some(acc) = state_changeset.get(&CHEATCODE_ADDRESS) {
526            if let Some(failed_slot) = acc.storage.get(&GLOBAL_FAIL_SLOT) {
527                if !failed_slot.present_value().is_zero() {
528                    return false;
529                }
530            }
531        }
532        if let Ok(failed_slot) = self.backend().storage_ref(CHEATCODE_ADDRESS, GLOBAL_FAIL_SLOT) {
533            if !failed_slot.is_zero() {
534                return false;
535            }
536        }
537
538        if !self.legacy_assertions {
539            return true;
540        }
541
542        // Finally, resort to calling `DSTest::failed`.
543        {
544            // Construct a new bare-bones backend to evaluate success.
545            let mut backend = self.backend().clone_empty();
546
547            // We only clone the test contract and cheatcode accounts,
548            // that's all we need to evaluate success.
549            for address in [address, CHEATCODE_ADDRESS] {
550                let Ok(acc) = self.backend().basic_ref(address) else { return false };
551                backend.insert_account_info(address, acc.unwrap_or_default());
552            }
553
554            // If this test failed any asserts, then this changeset will contain changes
555            // `false -> true` for the contract's `failed` variable and the `globalFailure` flag
556            // in the state of the cheatcode address,
557            // which are both read when we call `"failed()(bool)"` in the next step.
558            backend.commit(state_changeset.into_owned());
559
560            // Check if a DSTest assertion failed
561            let executor = self.clone_with_backend(backend);
562            let call = executor.call_sol(CALLER, address, &ITest::failedCall {}, U256::ZERO, None);
563            match call {
564                Ok(CallResult { raw: _, decoded_result: ITest::failedReturn { failed } }) => {
565                    trace!(failed, "DSTest::failed()");
566                    !failed
567                }
568                Err(err) => {
569                    trace!(%err, "failed to call DSTest::failed()");
570                    true
571                }
572            }
573        }
574    }

576    /// Creates the environment to use when executing a transaction in a test context
577    ///
578    /// If using a backend with cheatcodes, `tx.gas_price` and `block.number` will be overwritten by
579    /// the cheatcode state in between calls.
580    fn build_test_env(
581        &self,
582        caller: Address,
583        transact_to: TxKind,
584        data: Bytes,
585        value: U256,
586    ) -> EnvWithHandlerCfg {
587        let env = Env {
588            cfg: self.env().cfg.clone(),
589            // We always set the gas price to 0 so we can execute the transaction regardless of
590            // network conditions - the actual gas price is kept in `self.block` and is applied by
591            // the cheatcode handler if it is enabled
592            block: BlockEnv {
593                basefee: U256::ZERO,
594                gas_limit: U256::from(self.gas_limit),
595                ..self.env().block.clone()
596            },
597            tx: TxEnv {
598                caller,
599                transact_to,
600                data,
601                value,
602                // As above, we set the gas price to 0.
603                gas_price: U256::ZERO,
604                gas_priority_fee: None,
605                gas_limit: self.gas_limit,
606                ..self.env().tx.clone()
607            },
608        };
609
610        EnvWithHandlerCfg::new_with_spec_id(Box::new(env), self.spec_id())
611    }

613    pub fn call_sol_default<C: SolCall>(&self, to: Address, args: &C) -> C::Return
614    where
615        C::Return: Default,
616    {
617        self.call_sol(CALLER, to, args, U256::ZERO, None)
618            .map(|c| c.decoded_result)
619            .inspect_err(|e| warn!(target: "forge::test", "failed calling {:?}: {e}", C::SIGNATURE))
620            .unwrap_or_default()
621    }
622}

769impl RawCallResult {
770    /// Converts the result of the call into an `EvmError`.
771    pub fn into_evm_error(self, rd: Option<&RevertDecoder>) -> EvmError {
772        if let Some(reason) = SkipReason::decode(&self.result) {
773            return EvmError::Skip(reason);
774        }
775        let reason = rd.unwrap_or_default().decode(&self.result, Some(self.exit_reason));
776        EvmError::Execution(Box::new(self.into_execution_error(reason)))
777    }
778
779    /// Converts the result of the call into an `ExecutionErr`.
780    pub fn into_execution_error(self, reason: String) -> ExecutionErr {
781        ExecutionErr { raw: self, reason }
782    }
783
784    /// Returns an `EvmError` if the call failed, otherwise returns `self`.
785    pub fn into_result(self, rd: Option<&RevertDecoder>) -> Result<Self, EvmError> {
786        if self.exit_reason.is_ok() {
787            Ok(self)
788        } else {
789            Err(self.into_evm_error(rd))
790        }
791    }
792
793    /// Decodes the result of the call with the given function.
794    pub fn into_decoded_result(
795        mut self,
796        func: &Function,
797        rd: Option<&RevertDecoder>,
798    ) -> Result<CallResult, EvmError> {
799        self = self.into_result(rd)?;
800        let mut result = func.abi_decode_output(&self.result, false)?;
801        let decoded_result = if result.len() == 1 {
802            result.pop().unwrap()
803        } else {
804            // combine results into a tuple
805            DynSolValue::Tuple(result)
806        };
807        Ok(CallResult { raw: self, decoded_result })
808    }
809
810    /// Returns the transactions generated from this call.
811    pub fn transactions(&self) -> Option<&BroadcastableTransactions> {
812        self.cheatcodes.as_ref().map(|c| &c.broadcastable_transactions)
813    }
814}

840/// Converts the data aggregated in the `inspector` and `call` to a `RawCallResult`
841fn convert_executed_result(
842    env: EnvWithHandlerCfg,
843    inspector: InspectorStack,
844    ResultAndState { result, state: state_changeset }: ResultAndState,
845    has_state_snapshot_failure: bool,
846) -> eyre::Result<RawCallResult> {
847    let (exit_reason, gas_refunded, gas_used, out, exec_logs) = match result {
848        ExecutionResult::Success { reason, gas_used, gas_refunded, output, logs, .. } => {
849            (reason.into(), gas_refunded, gas_used, Some(output), logs)
850        }
851        ExecutionResult::Revert { gas_used, output } => {
852            // Need to fetch the unused gas
853            (InstructionResult::Revert, 0_u64, gas_used, Some(Output::Call(output)), vec![])
854        }
855        ExecutionResult::Halt { reason, gas_used } => {
856            (reason.into(), 0_u64, gas_used, None, vec![])
857        }
858    };
859    let stipend = revm::interpreter::gas::validate_initial_tx_gas(
860        env.spec_id(),
861        &env.tx.data,
862        env.tx.transact_to.is_create(),
863        &env.tx.access_list,
864        0,
865    );
866
867    let result = match &out {
868        Some(Output::Call(data)) => data.clone(),
869        _ => Bytes::new(),
870    };
871
872    let InspectorData { mut logs, labels, traces, coverage, cheatcodes, chisel_state } =
873        inspector.collect();
874
875    if logs.is_empty() {
876        logs = exec_logs;
877    }
878
879    let transactions = cheatcodes
880        .as_ref()
881        .map(|c| c.broadcastable_transactions.clone())
882        .filter(|txs| !txs.is_empty());
883
884    Ok(RawCallResult {
885        exit_reason,
886        reverted: !matches!(exit_reason, return_ok!()),
887        has_state_snapshot_failure,
888        result,
889        gas_used,
890        gas_refunded,
891        stipend,
892        logs,
893        labels,
894        traces,
895        coverage,
896        transactions,
897        state_changeset,
898        env,
899        cheatcodes,
900        out,
901        chisel_state,
902    })
