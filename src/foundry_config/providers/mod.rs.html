13/// Generate warnings for unknown sections and deprecated keys
14pub struct WarningsProvider<P> {
15    provider: P,
16    profile: Profile,
17    old_warnings: Result<Vec<Warning>, Error>,
18}
19
20impl<P> WarningsProvider<P> {
21    const WARNINGS_KEY: &'static str = "__warnings";
22
23    /// Creates a new warnings provider.
24    pub fn new(
25        provider: P,
26        profile: impl Into<Profile>,
27        old_warnings: Result<Vec<Warning>, Error>,
28    ) -> Self {
29        Self { provider, profile: profile.into(), old_warnings }
30    }
31
32    /// Creates a new figment warnings provider.
33    pub fn for_figment(provider: P, figment: &Figment) -> Self {
34        let old_warnings = {
35            let warnings_res = figment.extract_inner(Self::WARNINGS_KEY);
36            if warnings_res.as_ref().err().map(|err| err.missing()).unwrap_or(false) {
37                Ok(vec![])
38            } else {
39                warnings_res
40            }
41        };
42        Self::new(provider, figment.profile().clone(), old_warnings)
43    }
44}
45
46impl<P: Provider> WarningsProvider<P> {
47    /// Collects all warnings.
48    pub fn collect_warnings(&self) -> Result<Vec<Warning>, Error> {
49        let data = self.provider.data().unwrap_or_default();
50
51        let mut out = self.old_warnings.clone()?;
52
53        // Add warning for unknown sections.
54        out.extend(
55            data.keys()
56                .filter(|k| {
57                    **k != Config::PROFILE_SECTION &&
58                        !Config::STANDALONE_SECTIONS.iter().any(|s| s == k)
59                })
60                .map(|unknown_section| {
61                    let source = self.provider.metadata().source.map(|s| s.to_string());
62                    Warning::UnknownSection { unknown_section: unknown_section.clone(), source }
63                }),
64        );
65
66        // Add warning for deprecated keys.
67        let deprecated_key_warning = |key| {
68            DEPRECATIONS.iter().find_map(|(deprecated_key, new_value)| {
69                if key == *deprecated_key {
70                    Some(Warning::DeprecatedKey {
71                        old: deprecated_key.to_string(),
72                        new: new_value.to_string(),
73                    })
74                } else {
75                    None
76                }
77            })
78        };
79        let profiles = data
80            .iter()
81            .filter(|(profile, _)| **profile == Config::PROFILE_SECTION)
82            .map(|(_, dict)| dict);
83        out.extend(profiles.clone().flat_map(BTreeMap::keys).filter_map(deprecated_key_warning));
84        out.extend(
85            profiles
86                .filter_map(|dict| dict.get(self.profile.as_str().as_str()))
87                .filter_map(Value::as_dict)
88                .flat_map(BTreeMap::keys)
89                .filter_map(deprecated_key_warning),
90        );
91
92        Ok(out)
93    }
94}

96impl<P: Provider> Provider for WarningsProvider<P> {
97    fn metadata(&self) -> Metadata {
98        if let Some(source) = self.provider.metadata().source {
99            Metadata::from("Warnings", source)
100        } else {
101            Metadata::named("Warnings")
102        }
103    }
104
105    fn data(&self) -> Result<Map<Profile, Dict>, Error> {
106        Ok(Map::from([(
107            self.profile.clone(),
108            Dict::from([(
109                Self::WARNINGS_KEY.to_string(),
110                Value::serialize(self.collect_warnings()?)?,
111            )]),
112        )]))
113    }
114
115    fn profile(&self) -> Option<Profile> {
116        Some(self.profile.clone())
117    }
118}

120/// Extracts the profile from the `profile` key and sets unset values according to the fallback
121/// provider
122pub struct FallbackProfileProvider<P> {
123    provider: P,
124    profile: Profile,
125    fallback: Profile,
126}
127
128impl<P> FallbackProfileProvider<P> {
129    /// Creates a new fallback profile provider.
130    pub fn new(provider: P, profile: impl Into<Profile>, fallback: impl Into<Profile>) -> Self {
131        Self { provider, profile: profile.into(), fallback: fallback.into() }
132    }
133}
134
135impl<P: Provider> Provider for FallbackProfileProvider<P> {
136    fn metadata(&self) -> Metadata {
137        self.provider.metadata()
138    }
139
140    fn data(&self) -> Result<Map<Profile, Dict>, Error> {
141        if let Some(fallback) = self.provider.data()?.get(&self.fallback) {
142            let mut inner = self.provider.data()?.remove(&self.profile).unwrap_or_default();
143            for (k, v) in fallback.iter() {
144                if !inner.contains_key(k) {
145                    inner.insert(k.to_owned(), v.clone());
146                }
147            }
148            Ok(self.profile.collect(inner))
149        } else {
150            self.provider.data()
151        }
152    }
153
154    fn profile(&self) -> Option<Profile> {
155        Some(self.profile.clone())
156    }
