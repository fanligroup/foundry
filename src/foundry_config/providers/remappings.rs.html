14/// Wrapper types over a `Vec<Remapping>` that only appends unique remappings.
15#[derive(Clone, Debug, Default)]
16pub struct Remappings {
17    /// Remappings.
18    remappings: Vec<Remapping>,
19}
20
21impl Remappings {
22    /// Create a new `Remappings` wrapper with an empty vector.
23    pub fn new() -> Self {
24        Self { remappings: Vec::new() }
25    }
26
27    /// Create a new `Remappings` wrapper with a vector of remappings.
28    pub fn new_with_remappings(remappings: Vec<Remapping>) -> Self {
29        Self { remappings }
30    }
31
32    /// Filters the remappings vector by name and context.
33    fn filter_key(r: &Remapping) -> String {
34        match &r.context {
35            Some(str) => str.clone() + &r.name.clone(),
36            None => r.name.clone(),
37        }
38    }
39
40    /// Consumes the wrapper and returns the inner remappings vector.
41    pub fn into_inner(self) -> Vec<Remapping> {
42        let mut tmp = HashSet::new();
43        let remappings =
44            self.remappings.iter().filter(|r| tmp.insert(Self::filter_key(r))).cloned().collect();
45        remappings
46    }
47
48    /// Push an element to the remappings vector, but only if it's not already present.
49    pub fn push(&mut self, remapping: Remapping) {
50        if !self.remappings.iter().any(|existing| {
51            // What we're doing here is filtering for ambiguous paths. For example, if we have
52            // @prb/math/=node_modules/@prb/math/src/ as existing, and
53            // @prb/=node_modules/@prb/  as the one being checked,
54            // we want to keep the already existing one, which is the first one. This way we avoid
55            // having to deal with ambiguous paths which is unwanted when autodetecting remappings.
56            existing.name.starts_with(&remapping.name) && existing.context == remapping.context
57        }) {
58            self.remappings.push(remapping)
59        }
60    }
61
62    /// Extend the remappings vector, leaving out the remappings that are already present.
63    pub fn extend(&mut self, remappings: Vec<Remapping>) {
64        for remapping in remappings {
65            self.push(remapping);
66        }
67    }
68}

70/// A figment provider that checks if the remappings were previously set and if they're unset looks
71/// up the fs via
72///   - `DAPP_REMAPPINGS` || `FOUNDRY_REMAPPINGS` env var
73///   - `<root>/remappings.txt` file
74///   - `Remapping::find_many`.
75pub struct RemappingsProvider<'a> {
76    /// Whether to auto detect remappings from the `lib_paths`
77    pub auto_detect_remappings: bool,
78    /// The lib/dependency directories to scan for remappings
79    pub lib_paths: Cow<'a, Vec<PathBuf>>,
80    /// the root path used to turn an absolute `Remapping`, as we're getting it from
81    /// `Remapping::find_many` into a relative one.
82    pub root: &'a PathBuf,
83    /// This contains either:
84    ///   - previously set remappings
85    ///   - a `MissingField` error, which means previous provider didn't set the "remappings" field
86    ///   - other error, like formatting
87    pub remappings: Result<Vec<Remapping>, Error>,
88}
89
90impl<'a> RemappingsProvider<'a> {
91    /// Find and parse remappings for the projects
92    ///
93    /// **Order**
94    ///
95    /// Remappings are built in this order (last item takes precedence)
96    /// - Autogenerated remappings
97    /// - toml remappings
98    /// - `remappings.txt`
99    /// - Environment variables
100    /// - CLI parameters
101    fn get_remappings(&self, remappings: Vec<Remapping>) -> Result<Vec<Remapping>, Error> {
102        trace!("get all remappings from {:?}", self.root);
103        /// prioritizes remappings that are closer: shorter `path`
104        ///   - ("a", "1/2") over ("a", "1/2/3")
105        ///
106        /// grouped by remapping context
107        fn insert_closest(
108            mappings: &mut BTreeMap<Option<String>, BTreeMap<String, PathBuf>>,
109            context: Option<String>,
110            key: String,
111            path: PathBuf,
112        ) {
113            let context_mappings = mappings.entry(context).or_default();
114            match context_mappings.entry(key) {
115                Entry::Occupied(mut e) => {
116                    if e.get().components().count() > path.components().count() {
117                        e.insert(path);
118                    }
119                }
120                Entry::Vacant(e) => {
121                    e.insert(path);
122                }
123            }
124        }
125
126        // Let's first just extend the remappings with the ones that were passed in,
127        // without any filtering.
128        let mut user_remappings = Vec::new();
129
130        // check env vars
131        if let Some(env_remappings) = remappings_from_env_var("DAPP_REMAPPINGS")
132            .or_else(|| remappings_from_env_var("FOUNDRY_REMAPPINGS"))
133        {
134            user_remappings
135                .extend(env_remappings.map_err::<Error, _>(|err| err.to_string().into())?);
136        }
137
138        // check remappings.txt file
139        let remappings_file = self.root.join("remappings.txt");
140        if remappings_file.is_file() {
141            let content = fs::read_to_string(remappings_file).map_err(|err| err.to_string())?;
142            let remappings_from_file: Result<Vec<_>, _> =
143                remappings_from_newline(&content).collect();
144            user_remappings
145                .extend(remappings_from_file.map_err::<Error, _>(|err| err.to_string().into())?);
146        }
147
148        user_remappings.extend(remappings);
149        // Let's now use the wrapper to conditionally extend the remappings with the autodetected
150        // ones. We want to avoid duplicates, and the wrapper will handle this for us.
151        let mut all_remappings = Remappings::new_with_remappings(user_remappings);
152
153        // scan all library dirs and autodetect remappings
154        // TODO: if a lib specifies contexts for remappings manually, we need to figure out how to
155        // resolve that
156        if self.auto_detect_remappings {
157            let mut lib_remappings = BTreeMap::new();
158            // find all remappings of from libs that use a foundry.toml
159            for r in self.lib_foundry_toml_remappings() {
160                insert_closest(&mut lib_remappings, r.context, r.name, r.path.into());
161            }
162            // use auto detection for all libs
163            for r in self
164                .lib_paths
165                .iter()
166                .map(|lib| self.root.join(lib))
167                .inspect(|lib| trace!(?lib, "find all remappings"))
168                .flat_map(|lib| Remapping::find_many(&lib))
169            {
170                // this is an additional safety check for weird auto-detected remappings
171                if ["lib/", "src/", "contracts/"].contains(&r.name.as_str()) {
172                    trace!(target: "forge", "- skipping the remapping");
173                    continue
174                }
175                insert_closest(&mut lib_remappings, r.context, r.name, r.path.into());
176            }
177
178            all_remappings.extend(
179                lib_remappings
180                    .into_iter()
181                    .flat_map(|(context, remappings)| {
182                        remappings.into_iter().map(move |(name, path)| Remapping {
183                            context: context.clone(),
184                            name,
185                            path: path.to_string_lossy().into(),
186                        })
187                    })
188                    .collect(),
189            );
190        }
191
192        Ok(all_remappings.into_inner())
193    }
194
195    /// Returns all remappings declared in foundry.toml files of libraries
196    fn lib_foundry_toml_remappings(&self) -> impl Iterator<Item = Remapping> + '_ {
197        self.lib_paths
198            .iter()
199            .map(|p| self.root.join(p))
200            .flat_map(foundry_toml_dirs)
201            .inspect(|lib| {
202                trace!("find all remappings of nested foundry.toml lib: {:?}", lib);
203            })
204            .flat_map(|lib: PathBuf| {
205                // load config, of the nested lib if it exists
206                let config = Config::load_with_root(&lib).sanitized();
207
208                // if the configured _src_ directory is set to something that
209                // [Remapping::find_many()] doesn't classify as a src directory (src, contracts,
210                // lib), then we need to manually add a remapping here
211                let mut src_remapping = None;
212                if ![Path::new("src"), Path::new("contracts"), Path::new("lib")]
213                    .contains(&config.src.as_path())
214                {
215                    if let Some(name) = lib.file_name().and_then(|s| s.to_str()) {
216                        let mut r = Remapping {
217                            context: None,
218                            name: format!("{name}/"),
219                            path: format!("{}", lib.join(&config.src).display()),
220                        };
221                        if !r.path.ends_with('/') {
222                            r.path.push('/')
223                        }
224                        src_remapping = Some(r);
225                    }
226                }
227
228                // Eventually, we could set context for remappings at this location,
229                // taking into account the OS platform. We'll need to be able to handle nested
230                // contexts depending on dependencies for this to work.
231                // For now, we just leave the default context (none).
232                let mut remappings =
233                    config.remappings.into_iter().map(Remapping::from).collect::<Vec<Remapping>>();
234
235                if let Some(r) = src_remapping {
236                    remappings.push(r);
237                }
238                remappings
239            })
240    }
241}

243impl<'a> Provider for RemappingsProvider<'a> {
244    fn metadata(&self) -> Metadata {
245        Metadata::named("Remapping Provider")
246    }
247
248    fn data(&self) -> Result<Map<Profile, Dict>, Error> {
249        let remappings = match &self.remappings {
250            Ok(remappings) => self.get_remappings(remappings.clone()),
251            Err(err) => {
252                if let figment::error::Kind::MissingField(_) = err.kind {
253                    self.get_remappings(vec![])
254                } else {
255                    return Err(err.clone())
256                }
257            }
258        }?;
259
260        // turn the absolute remapping into a relative one by stripping the `root`
261        let remappings = remappings
262            .into_iter()
263            .map(|r| RelativeRemapping::new(r, self.root).to_string())
264            .collect::<Vec<_>>();
265
266        Ok(Map::from([(
267            Config::selected_profile(),
268            Dict::from([("remappings".to_string(), figment::value::Value::from(remappings))]),
269        )]))
270    }
271
272    fn profile(&self) -> Option<Profile> {
273        Some(Config::selected_profile())
274    }
