24/// Loads the config for the current project workspace or the provided root path.
25///
26/// # Panics
27///
28/// Panics if the project root cannot be found. See [`find_project_root`].
29#[track_caller]
30pub fn load_config_with_root(root: Option<&Path>) -> Config {
31    let root = match root {
32        Some(root) => root,
33        None => &find_project_root(None),
34    };
35    Config::load_with_root(root).sanitized()
36}

38/// Returns the path of the top-level directory of the working git tree.
39pub fn find_git_root(relative_to: &Path) -> io::Result<Option<PathBuf>> {
40    let root =
41        if relative_to.is_absolute() { relative_to } else { &dunce::canonicalize(relative_to)? };
42    Ok(root.ancestors().find(|p| p.join(".git").is_dir()).map(Path::to_path_buf))
43}

45/// Returns the root path to set for the project root.
46///
47/// Traverse the dir tree up and look for a `foundry.toml` file starting at the given path or cwd,
48/// but only until the root dir of the current repo so that:
49///
50/// ```text
51/// -- foundry.toml
52///
53/// -- repo
54///   |__ .git
55///   |__sub
56///      |__ [given_path | cwd]
57/// ```
58///
59/// will still detect `repo` as root.
60///
61/// Returns `repo` or `cwd` if no `foundry.toml` is found in the tree.
62///
63/// # Panics
64///
65/// Panics if:
66/// - `cwd` is `Some` and is not a valid directory;
67/// - `cwd` is `None` and the [`std::env::current_dir`] call fails.
68#[track_caller]
69pub fn find_project_root(cwd: Option<&Path>) -> PathBuf {
70    try_find_project_root(cwd).expect("Could not find project root")
71}
72
73/// Returns the root path to set for the project root.
74///
75/// Same as [`find_project_root`], but returns an error instead of panicking.
76pub fn try_find_project_root(cwd: Option<&Path>) -> io::Result<PathBuf> {
77    let cwd = match cwd {
78        Some(path) => path,
79        None => &std::env::current_dir()?,
80    };
81    let boundary = find_git_root(cwd)?;
82    let found = cwd
83        .ancestors()
84        // Don't look outside of the git repo if it exists.
85        .take_while(|p| if let Some(boundary) = &boundary { p.starts_with(boundary) } else { true })
86        .find(|p| p.join(Config::FILE_NAME).is_file())
87        .map(Path::to_path_buf);
88    Ok(found.or(boundary).unwrap_or_else(|| cwd.to_path_buf()))
89}

91/// Returns all [`Remapping`]s contained in the `remappings` str separated by newlines
92///
93/// # Example
94///
95/// ```
96/// use foundry_config::remappings_from_newline;
97/// let remappings: Result<Vec<_>, _> = remappings_from_newline(
98///     r#"
99///              file-ds-test/=lib/ds-test/
100///              file-other/=lib/other/
101///          "#,
102/// )
103/// .collect();
104/// ```
105pub fn remappings_from_newline(
106    remappings: &str,
107) -> impl Iterator<Item = Result<Remapping, RemappingError>> + '_ {
108    remappings.lines().map(|x| x.trim()).filter(|x| !x.is_empty()).map(Remapping::from_str)
109}

111/// Returns the remappings from the given var
112///
113/// Returns `None` if the env var is not set, otherwise all Remappings, See
114/// `remappings_from_newline`
115pub fn remappings_from_env_var(env_var: &str) -> Option<Result<Vec<Remapping>, RemappingError>> {
116    let val = std::env::var(env_var).ok()?;
117    Some(remappings_from_newline(&val).collect())
118}

120/// Converts the `val` into a `figment::Value::Array`
121///
122/// The values should be separated by commas, surrounding brackets are also supported `[a,b,c]`
123pub fn to_array_value(val: &str) -> Result<Value, figment::Error> {
124    let value: Value = match Value::from(val) {
125        Value::String(_, val) => val
126            .trim_start_matches('[')
127            .trim_end_matches(']')
128            .split(',')
129            .map(|s| s.to_string())
130            .collect::<Vec<_>>()
131            .into(),
132        Value::Empty(_, _) => Vec::<Value>::new().into(),
133        val @ Value::Array(_, _) => val,
134        _ => return Err(format!("Invalid value `{val}`, expected an array").into()),
135    };
136    Ok(value)
137}

139/// Returns a list of _unique_ paths to all folders under `root` that contain a `foundry.toml` file
140///
141/// This will also resolve symlinks
142///
143/// # Example
144///
145/// ```no_run
146/// use foundry_config::utils;
147/// let dirs = utils::foundry_toml_dirs("./lib");
148/// ```
149///
150/// for following layout this will return
151/// `["lib/dep1", "lib/dep2"]`
152///
153/// ```text
154/// lib
155/// └── dep1
156/// │   ├── foundry.toml
157/// └── dep2
158///     ├── foundry.toml
159/// ```
160pub fn foundry_toml_dirs(root: impl AsRef<Path>) -> Vec<PathBuf> {
161    walkdir::WalkDir::new(root)
162        .max_depth(1)
163        .into_iter()
164        .filter_map(Result::ok)
165        .filter(|e| e.file_type().is_dir())
166        .filter_map(|e| dunce::canonicalize(e.path()).ok())
167        .filter(|p| p.join(Config::FILE_NAME).exists())
168        .collect()
169}

171/// Returns a remapping for the given dir
172pub(crate) fn get_dir_remapping(dir: impl AsRef<Path>) -> Option<Remapping> {
173    let dir = dir.as_ref();
174    if let Some(dir_name) = dir.file_name().and_then(|s| s.to_str()).filter(|s| !s.is_empty()) {
175        let mut r = Remapping {
176            context: None,
177            name: format!("{dir_name}/"),
178            path: format!("{}", dir.display()),
179        };
180        if !r.path.ends_with('/') {
181            r.path.push('/')
182        }
183        Some(r)
184    } else {
185        None
186    }
187}

189/// Returns all available `profile` keys in a given `.toml` file
190///
191/// i.e. The toml below would return would return `["default", "ci", "local"]`
192/// ```toml
193/// [profile.default]
194/// ...
195/// [profile.ci]
196/// ...
197/// [profile.local]
198/// ```
199pub fn get_available_profiles(toml_path: impl AsRef<Path>) -> eyre::Result<Vec<String>> {
200    let mut result = vec![Config::DEFAULT_PROFILE.to_string()];
201
202    if !toml_path.as_ref().exists() {
203        return Ok(result)
204    }
205
206    let doc = read_toml(toml_path)?;
207
208    if let Some(Item::Table(profiles)) = doc.as_table().get(Config::PROFILE_SECTION) {
209        for (profile, _) in profiles {
210            let p = profile.to_string();
211            if !result.contains(&p) {
212                result.push(p);
213            }
214        }
215    }
216
217    Ok(result)
218}

220/// Returns a [`toml_edit::Document`] loaded from the provided `path`.
221/// Can raise an error in case of I/O or parsing errors.
222fn read_toml(path: impl AsRef<Path>) -> eyre::Result<DocumentMut> {
223    let path = path.as_ref().to_owned();
224    let doc: DocumentMut = std::fs::read_to_string(path)?.parse()?;
225    Ok(doc)
226}

228/// Deserialize stringified percent. The value must be between 0 and 100 inclusive.
229pub(crate) fn deserialize_stringified_percent<'de, D>(deserializer: D) -> Result<u32, D::Error>
230where
231    D: Deserializer<'de>,
232{
233    let num: U256 = Numeric::deserialize(deserializer)?.into();
234    let num: u64 = num.try_into().map_err(serde::de::Error::custom)?;
235    if num <= 100 {
236        num.try_into().map_err(serde::de::Error::custom)
237    } else {
238        Err(serde::de::Error::custom("percent must be lte 100"))
239    }
240}

242/// Deserialize a `u64` or "max" for `u64::MAX`.
243pub(crate) fn deserialize_u64_or_max<'de, D>(deserializer: D) -> Result<u64, D::Error>
244where
245    D: Deserializer<'de>,
246{
247    #[derive(Deserialize)]
248    #[serde(untagged)]
249    enum Val {
250        Number(u64),
251        String(String),
252    }
253
254    match Val::deserialize(deserializer)? {
255        Val::Number(num) => Ok(num),
256        Val::String(s) if s.eq_ignore_ascii_case("max") => Ok(u64::MAX),
257        Val::String(s) => s.parse::<u64>().map_err(D::Error::custom),
258    }
259}
260
261/// Deserialize a `usize` or "max" for `usize::MAX`.
262pub(crate) fn deserialize_usize_or_max<'de, D>(deserializer: D) -> Result<usize, D::Error>
263where
264    D: Deserializer<'de>,
265{
266    deserialize_u64_or_max(deserializer)?.try_into().map_err(D::Error::custom)
267}

269/// Helper type to parse both `u64` and `U256`
270#[derive(Clone, Copy, Deserialize)]
271#[serde(untagged)]
272pub enum Numeric {
273    /// A [U256] value.
274    U256(U256),
275    /// A `u64` value.
276    Num(u64),
277}
278
279impl From<Numeric> for U256 {
280    fn from(n: Numeric) -> Self {
281        match n {
282            Numeric::U256(n) => n,
283            Numeric::Num(n) => Self::from(n),
284        }
285    }
286}
287
288impl FromStr for Numeric {
289    type Err = String;
290
291    fn from_str(s: &str) -> Result<Self, Self::Err> {
292        if s.starts_with("0x") {
293            U256::from_str_radix(s, 16).map(Numeric::U256).map_err(|err| err.to_string())
294        } else {
295            U256::from_str(s).map(Numeric::U256).map_err(|err| err.to_string())
296        }
297    }
298}

300/// Returns the [SpecId] derived from [EvmVersion]
301#[inline]
302pub fn evm_spec_id(evm_version: &EvmVersion, alphanet: bool) -> SpecId {
303    if alphanet {
304        return SpecId::PRAGUE_EOF;
305    }
306    match evm_version {
307        EvmVersion::Homestead => SpecId::HOMESTEAD,
308        EvmVersion::TangerineWhistle => SpecId::TANGERINE,
309        EvmVersion::SpuriousDragon => SpecId::SPURIOUS_DRAGON,
310        EvmVersion::Byzantium => SpecId::BYZANTIUM,
311        EvmVersion::Constantinople => SpecId::CONSTANTINOPLE,
312        EvmVersion::Petersburg => SpecId::PETERSBURG,
313        EvmVersion::Istanbul => SpecId::ISTANBUL,
314        EvmVersion::Berlin => SpecId::BERLIN,
315        EvmVersion::London => SpecId::LONDON,
316        EvmVersion::Paris => SpecId::MERGE,
317        EvmVersion::Shanghai => SpecId::SHANGHAI,
318        EvmVersion::Cancun => SpecId::CANCUN,
319        EvmVersion::Prague => SpecId::PRAGUE_EOF,
320    }
321}
