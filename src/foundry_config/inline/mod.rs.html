22/// Represents per-test configurations, declared inline
23/// as structured comments in Solidity test files. This allows
24/// to create configs directly bound to a solidity test.
25#[derive(Clone, Debug, Default)]
26pub struct InlineConfig<T> {
27    /// Contract-level configurations, used for functions that do not have a specific
28    /// configuration.
29    contract_level: HashMap<String, T>,
30    /// Maps a (test-contract, test-function) pair
31    /// to a specific configuration provided by the user.
32    fn_level: HashMap<(String, String), T>,
33}
34
35impl<T> InlineConfig<T> {
36    /// Returns an inline configuration, if any, for a test function.
37    /// Configuration is identified by the pair "contract", "function".
38    pub fn get(&self, contract_id: &str, fn_name: &str) -> Option<&T> {
39        let key = (contract_id.to_string(), fn_name.to_string());
40        self.fn_level.get(&key).or_else(|| self.contract_level.get(contract_id))
41    }
42
43    pub fn insert_contract(&mut self, contract_id: impl Into<String>, config: T) {
44        self.contract_level.insert(contract_id.into(), config);
45    }
46
47    /// Inserts an inline configuration, for a test function.
48    /// Configuration is identified by the pair "contract", "function".
49    pub fn insert_fn<C, F>(&mut self, contract_id: C, fn_name: F, config: T)
50    where
51        C: Into<String>,
52        F: Into<String>,
53    {
54        let key = (contract_id.into(), fn_name.into());
55        self.fn_level.insert(key, config);
56    }
57}
