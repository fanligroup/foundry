5/// This trait is intended to parse configurations from
6/// structured text. Foundry users can annotate Solidity test functions,
7/// providing special configs just for the execution of a specific test.
8///
9/// An example:
10///
11/// ```solidity
12/// contract MyTest is Test {
13/// /// forge-config: default.fuzz.runs = 100
14/// /// forge-config: ci.fuzz.runs = 500
15/// function test_SimpleFuzzTest(uint256 x) public {...}
16///
17/// /// forge-config: default.fuzz.runs = 500
18/// /// forge-config: ci.fuzz.runs = 10000
19/// function test_ImportantFuzzTest(uint256 x) public {...}
20/// }
21/// ```
22pub trait InlineConfigParser
23where
24    Self: Clone + Default + Sized + 'static,
25{
26    /// Returns a config key that is common to all valid configuration lines
27    /// for the current impl. This helps to extract correct values out of a text.
28    ///
29    /// An example key would be `fuzz` of `invariant`.
30    fn config_key() -> String;
31
32    /// Tries to override `self` properties with values specified in the `configs` parameter.
33    ///
34    /// Returns
35    /// - `Some(Self)` in case some configurations are merged into self.
36    /// - `None` in case there are no configurations that can be applied to self.
37    /// - `Err(InlineConfigParserError)` in case of wrong configuration.
38    fn try_merge(&self, configs: &[String]) -> Result<Option<Self>, InlineConfigParserError>;
39
40    /// Validates and merges the natspec configs for current profile into the current config.
41    fn merge(&self, natspec: &NatSpec) -> Result<Option<Self>, InlineConfigError> {
42        let config_key = Self::config_key();
43
44        let configs = natspec
45            .current_profile_configs()
46            .filter(|l| l.contains(&config_key))
47            .collect::<Vec<String>>();
48
49        self.try_merge(&configs).map_err(|e| {
50            let line = natspec.debug_context();
51            InlineConfigError { line, source: e }
52        })
53    }
54
55    /// Given a list of config lines, returns all available pairs (key, value) matching the current
56    /// config key.
57    ///
58    /// # Examples
59    ///
60    /// ```ignore
61    /// assert_eq!(
62    ///     get_config_overrides(&[
63    ///         "forge-config: default.invariant.runs = 500",
64    ///         "forge-config: default.invariant.depth = 500",
65    ///         "forge-config: ci.invariant.depth = 500",
66    ///         "forge-config: ci.fuzz.runs = 10",
67    ///     ]),
68    ///     [("runs", "500"), ("depth", "500"), ("depth", "500")]
69    /// );
70    /// ```
71    fn get_config_overrides(config_lines: &[String]) -> Vec<(String, String)> {
72        let mut result: Vec<(String, String)> = vec![];
73        let config_key = Self::config_key();
74        let profile = ".*";
75        let prefix = format!("^{INLINE_CONFIG_PREFIX}:{profile}{config_key}\\.");
76        let re = Regex::new(&prefix).unwrap();
77
78        config_lines
79            .iter()
80            .map(|l| remove_whitespaces(l))
81            .filter(|l| re.is_match(l))
82            .map(|l| re.replace(&l, "").to_string())
83            .for_each(|line| {
84                let key_value = line.split('=').collect::<Vec<&str>>(); // i.e. "['runs', '500']"
85                if let Some(key) = key_value.first() {
86                    if let Some(value) = key_value.last() {
87                        result.push((key.to_string(), value.to_string()));
88                    }
89                }
90            });
91
92        result
93    }
94}

96/// Checks if all configuration lines specified in `natspec` use a valid profile.
97///
98/// i.e. Given available profiles
99/// ```rust
100/// let _profiles = vec!["ci", "default"];
101/// ```
102/// A configuration like `forge-config: ciii.invariant.depth = 1` would result
103/// in an error.
104pub fn validate_profiles(natspec: &NatSpec, profiles: &[String]) -> Result<(), InlineConfigError> {
105    for config in natspec.config_lines() {
106        if !profiles.iter().any(|p| config.starts_with(&format!("{INLINE_CONFIG_PREFIX}:{p}."))) {
107            let err_line: String = natspec.debug_context();
108            let profiles = format!("{profiles:?}");
109            Err(InlineConfigError {
110                source: InlineConfigParserError::InvalidProfile(config, profiles),
111                line: err_line,
112            })?
113        }
114    }
115    Ok(())
116}

118/// Tries to parse a `u32` from `value`. The `key` argument is used to give details
119/// in the case of an error.
120pub fn parse_config_u32(key: String, value: String) -> Result<u32, InlineConfigParserError> {
121    value.parse().map_err(|_| InlineConfigParserError::ParseInt(key, value))
122}
123
124/// Tries to parse a `bool` from `value`. The `key` argument is used to give details
125/// in the case of an error.
126pub fn parse_config_bool(key: String, value: String) -> Result<bool, InlineConfigParserError> {
127    value.parse().map_err(|_| InlineConfigParserError::ParseBool(key, value))
128}
