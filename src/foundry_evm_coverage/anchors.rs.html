8/// Attempts to find anchors for the given items using the given source map and bytecode.
9pub fn find_anchors(
10    bytecode: &[u8],
11    source_map: &SourceMap,
12    ic_pc_map: &IcPcMap,
13    items: &[CoverageItem],
14    items_by_source_id: &FxHashMap<usize, Vec<usize>>,
15) -> Vec<ItemAnchor> {
16    let mut seen = FxHashSet::default();
17    source_map
18        .iter()
19        .filter_map(|element| items_by_source_id.get(&(element.index()? as usize)))
20        .flatten()
21        .filter_map(|&item_id| {
22            if !seen.insert(item_id) {
23                return None;
24            }
25
26            let item = &items[item_id];
27            let find_anchor_by_first_opcode = |item: &CoverageItem| match find_anchor_simple(
28                source_map, ic_pc_map, item_id, &item.loc,
29            ) {
30                Ok(anchor) => Some(anchor),
31                Err(e) => {
32                    warn!("Could not find anchor for item {item}: {e}");
33                    None
34                }
35            };
36            match item.kind {
37                CoverageItemKind::Branch { path_id, is_first_opcode, .. } => {
38                    if is_first_opcode {
39                        find_anchor_by_first_opcode(item)
40                    } else {
41                        match find_anchor_branch(bytecode, source_map, item_id, &item.loc) {
42                            Ok(anchors) => match path_id {
43                                0 => Some(anchors.0),
44                                1 => Some(anchors.1),
45                                _ => panic!("Too many paths for branch"),
46                            },
47                            Err(e) => {
48                                warn!("Could not find anchor for item {item}: {e}");
49                                None
50                            }
51                        }
52                    }
53                }
54                _ => find_anchor_by_first_opcode(item),
55            }
56        })
57        .collect()
58}

60/// Find an anchor representing the first opcode within the given source range.
61pub fn find_anchor_simple(
62    source_map: &SourceMap,
63    ic_pc_map: &IcPcMap,
64    item_id: usize,
65    loc: &SourceLocation,
66) -> eyre::Result<ItemAnchor> {
67    let instruction =
68        source_map.iter().position(|element| is_in_source_range(element, loc)).ok_or_else(
69            || eyre::eyre!("Could not find anchor: No matching instruction in range {loc}"),
70        )?;
71
72    Ok(ItemAnchor {
73        instruction: ic_pc_map.get(instruction).ok_or_else(|| {
74            eyre::eyre!("We found an anchor, but we can't translate it to a program counter")
75        })?,
76        item_id,
77    })
78}

80/// Finds the anchor corresponding to a branch item.
81///
82/// This finds the relevant anchors for a branch coverage item. These anchors
83/// are found using the bytecode of the contract in the range of the branching node.
84///
85/// For `IfStatement` nodes, the template is generally:
86/// ```text
87/// <condition>
88/// PUSH <ic if false>
89/// JUMPI
90/// <true branch>
91/// <...>
92/// <false branch>
93/// ```
94///
95/// For `assert` and `require`, the template is generally:
96///
97/// ```text
98/// PUSH <ic if true>
99/// JUMPI
100/// <revert>
101/// <...>
102/// <true branch>
103/// ```
104///
105/// This function will look for the last JUMPI instruction, backtrack to find the program
106/// counter of the first branch, and return an item for that program counter, and the
107/// program counter immediately after the JUMPI instruction.
108pub fn find_anchor_branch(
109    bytecode: &[u8],
110    source_map: &SourceMap,
111    item_id: usize,
112    loc: &SourceLocation,
113) -> eyre::Result<(ItemAnchor, ItemAnchor)> {
114    // NOTE(onbjerg): We use `SpecId::LATEST` here since it does not matter; the only difference
115    // is the gas cost.
116
117    let mut anchors: Option<(ItemAnchor, ItemAnchor)> = None;
118    let mut pc = 0;
119    let mut cumulative_push_size = 0;
120    while pc < bytecode.len() {
121        let op = bytecode[pc];
122
123        // We found a push, so we do some PC -> IC translation accounting, but we also check if
124        // this push is coupled with the JUMPI we are interested in.
125
126        // Check if Opcode is PUSH
127        if (opcode::PUSH1..=opcode::PUSH32).contains(&op) {
128            let element = if let Some(element) = source_map.get(pc - cumulative_push_size) {
129                element
130            } else {
131                // NOTE(onbjerg): For some reason the last few bytes of the bytecode do not have
132                // a source map associated, so at that point we just stop searching
133                break
134            };
135
136            // Do push byte accounting
137            let push_size = (op - opcode::PUSH1 + 1) as usize;
138            pc += push_size;
139            cumulative_push_size += push_size;
140
141            // Check if we are in the source range we are interested in, and if the next opcode
142            // is a JUMPI
143            if is_in_source_range(element, loc) && bytecode[pc + 1] == opcode::JUMPI {
144                // We do not support program counters bigger than usize. This is also an
145                // assumption in REVM, so this is just a sanity check.
146                ensure!(push_size <= 8, "jump destination overflow");
147
148                // Convert the push bytes for the second branch's PC to a usize
149                let push_bytes_start = pc - push_size + 1;
150                let push_bytes = &bytecode[push_bytes_start..push_bytes_start + push_size];
151                let mut pc_bytes = [0u8; 8];
152                pc_bytes[8 - push_size..].copy_from_slice(push_bytes);
153                let pc_jump = u64::from_be_bytes(pc_bytes);
154                let pc_jump = usize::try_from(pc_jump).expect("PC is too big");
155                anchors = Some((
156                    ItemAnchor {
157                        item_id,
158                        // The first branch is the opcode directly after JUMPI
159                        instruction: pc + 2,
160                    },
161                    ItemAnchor { item_id, instruction: pc_jump },
162                ));
163            }
164        }
165        pc += 1;
166    }
167
168    anchors.ok_or_else(|| eyre::eyre!("Could not detect branches in source: {}", loc))
169}

171/// Calculates whether `element` is within the range of the target `location`.
172fn is_in_source_range(element: &SourceElement, location: &SourceLocation) -> bool {
173    // Source IDs must match.
174    let source_ids_match = element.index().map_or(false, |a| a as usize == location.source_id);
175    if !source_ids_match {
176        return false;
177    }
178
179    // Needed because some source ranges in the source map mark the entire contract...
180    let is_within_start = element.offset() >= location.start;
181    if !is_within_start {
182        return false;
183    }
184
185    let start_of_ranges = location.start.max(element.offset());
186    let end_of_ranges = (location.start + location.length.unwrap_or_default())
187        .min(element.offset() + element.length());
188    let within_ranges = start_of_ranges <= end_of_ranges;
189    if !within_ranges {
190        return false;
191    }
192
193    true
