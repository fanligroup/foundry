8/// A visitor that walks the AST of a single contract and finds coverage items.
9#[derive(Clone, Debug)]
10pub struct ContractVisitor<'a> {
11    /// The source ID of the contract.
12    source_id: usize,
13    /// The source code that contains the AST being walked.
14    source: &'a str,
15
16    /// The name of the contract being walked.
17    contract_name: &'a Arc<str>,
18
19    /// The current branch ID
20    branch_id: usize,
21    /// Stores the last line we put in the items collection to ensure we don't push duplicate lines
22    last_line: usize,
23
24    /// Coverage items
25    pub items: Vec<CoverageItem>,
26}
27
28impl<'a> ContractVisitor<'a> {
29    pub fn new(source_id: usize, source: &'a str, contract_name: &'a Arc<str>) -> Self {
30        Self { source_id, source, contract_name, branch_id: 0, last_line: 0, items: Vec::new() }
31    }
32
33    pub fn visit_contract(&mut self, node: &Node) -> eyre::Result<()> {
34        // Find all functions and walk their AST
35        for node in &node.nodes {
36            match node.node_type {
37                NodeType::FunctionDefinition => {
38                    self.visit_function_definition(node)?;
39                }
40                NodeType::ModifierDefinition => {
41                    self.visit_modifier_or_yul_fn_definition(node)?;
42                }
43                _ => {}
44            }
45        }
46        Ok(())
47    }
48
49    fn visit_function_definition(&mut self, node: &Node) -> eyre::Result<()> {
50        let name: String =
51            node.attribute("name").ok_or_else(|| eyre::eyre!("Function has no name"))?;
52
53        // TODO(onbjerg): Figure out why we cannot find anchors for the receive function
54        let kind: String =
55            node.attribute("kind").ok_or_else(|| eyre::eyre!("Function has no kind"))?;
56        if kind == "receive" {
57            return Ok(())
58        }
59
60        match &node.body {
61            Some(body) => {
62                self.push_item_kind(CoverageItemKind::Function { name }, &node.src);
63                self.visit_block(body)
64            }
65            _ => Ok(()),
66        }
67    }
68
69    fn visit_modifier_or_yul_fn_definition(&mut self, node: &Node) -> eyre::Result<()> {
70        let name: String =
71            node.attribute("name").ok_or_else(|| eyre::eyre!("Modifier has no name"))?;
72
73        match &node.body {
74            Some(body) => {
75                self.push_item_kind(CoverageItemKind::Function { name }, &node.src);
76                self.visit_block(body)
77            }
78            _ => Ok(()),
79        }
80    }
81
82    fn visit_block(&mut self, node: &Node) -> eyre::Result<()> {
83        let statements: Vec<Node> = node.attribute("statements").unwrap_or_default();
84
85        for statement in &statements {
86            self.visit_statement(statement)?;
87        }
88
89        Ok(())
90    }
91
92    fn visit_statement(&mut self, node: &Node) -> eyre::Result<()> {
93        match node.node_type {
94            // Blocks
95            NodeType::Block | NodeType::UncheckedBlock | NodeType::YulBlock => {
96                self.visit_block(node)
97            }
98            // Inline assembly block
99            NodeType::InlineAssembly => self.visit_block(
100                &node
101                    .attribute("AST")
102                    .ok_or_else(|| eyre::eyre!("inline assembly block with no AST attribute"))?,
103            ),
104            // Simple statements
105            NodeType::Break |
106            NodeType::Continue |
107            NodeType::EmitStatement |
108            NodeType::RevertStatement |
109            NodeType::YulAssignment |
110            NodeType::YulBreak |
111            NodeType::YulContinue |
112            NodeType::YulLeave |
113            NodeType::YulVariableDeclaration => {
114                self.push_item_kind(CoverageItemKind::Statement, &node.src);
115                Ok(())
116            }
117            // Skip placeholder statements as they are never referenced in source maps.
118            NodeType::PlaceholderStatement => Ok(()),
119            // Return with eventual subcall
120            NodeType::Return => {
121                self.push_item_kind(CoverageItemKind::Statement, &node.src);
122                if let Some(expr) = node.attribute("expression") {
123                    self.visit_expression(&expr)?;
124                }
125                Ok(())
126            }
127            // Variable declaration
128            NodeType::VariableDeclarationStatement => {
129                self.push_item_kind(CoverageItemKind::Statement, &node.src);
130                if let Some(expr) = node.attribute("initialValue") {
131                    self.visit_expression(&expr)?;
132                }
133                Ok(())
134            }
135            // While loops
136            NodeType::DoWhileStatement | NodeType::WhileStatement => {
137                self.visit_expression(
138                    &node
139                        .attribute("condition")
140                        .ok_or_else(|| eyre::eyre!("while statement had no condition"))?,
141                )?;
142
143                let body = node
144                    .body
145                    .as_deref()
146                    .ok_or_else(|| eyre::eyre!("while statement had no body node"))?;
147                self.visit_block_or_statement(body)
148            }
149            // For loops
150            NodeType::ForStatement => {
151                if let Some(stmt) = node.attribute("initializationExpression") {
152                    self.visit_statement(&stmt)?;
153                }
154                if let Some(expr) = node.attribute("condition") {
155                    self.visit_expression(&expr)?;
156                }
157                if let Some(stmt) = node.attribute("loopExpression") {
158                    self.visit_statement(&stmt)?;
159                }
160
161                let body = node
162                    .body
163                    .as_deref()
164                    .ok_or_else(|| eyre::eyre!("for statement had no body node"))?;
165                self.visit_block_or_statement(body)
166            }
167            // Expression statement
168            NodeType::ExpressionStatement | NodeType::YulExpressionStatement => self
169                .visit_expression(
170                    &node
171                        .attribute("expression")
172                        .ok_or_else(|| eyre::eyre!("expression statement had no expression"))?,
173                ),
174            // If statement
175            NodeType::IfStatement => {
176                self.visit_expression(
177                    &node
178                        .attribute("condition")
179                        .ok_or_else(|| eyre::eyre!("if statement had no condition"))?,
180                )?;
181
182                let true_body: Node = node
183                    .attribute("trueBody")
184                    .ok_or_else(|| eyre::eyre!("if statement had no true body"))?;
185
186                // We need to store the current branch ID here since visiting the body of either of
187                // the if blocks may increase `self.branch_id` in the case of nested if statements.
188                let branch_id = self.branch_id;
189
190                // We increase the branch ID here such that nested branches do not use the same
191                // branch ID as we do.
192                self.branch_id += 1;
193
194                match node.attribute::<Node>("falseBody") {
195                    // Both if/else statements.
196                    Some(false_body) => {
197                        // Add branch coverage items only if one of true/branch bodies contains
198                        // statements.
199                        if has_statements(&true_body) || has_statements(&false_body) {
200                            // The branch instruction is mapped to the first opcode within the true
201                            // body source range.
202                            self.push_item_kind(
203                                CoverageItemKind::Branch {
204                                    branch_id,
205                                    path_id: 0,
206                                    is_first_opcode: true,
207                                },
208                                &true_body.src,
209                            );
210                            // Add the coverage item for branch 1 (false body).
211                            // The relevant source range for the false branch is the `else`
212                            // statement itself and the false body of the else statement.
213                            self.push_item_kind(
214                                CoverageItemKind::Branch {
215                                    branch_id,
216                                    path_id: 1,
217                                    is_first_opcode: false,
218                                },
219                                &ast::LowFidelitySourceLocation {
220                                    start: node.src.start,
221                                    length: false_body.src.length.map(|length| {
222                                        false_body.src.start - true_body.src.start + length
223                                    }),
224                                    index: node.src.index,
225                                },
226                            );
227
228                            // Process the true body.
229                            self.visit_block_or_statement(&true_body)?;
230                            // Process the false body.
231                            self.visit_block_or_statement(&false_body)?;
232                        }
233                    }
234                    None => {
235                        // Add single branch coverage only if it contains statements.
236                        if has_statements(&true_body) {
237                            // Add the coverage item for branch 0 (true body).
238                            self.push_item_kind(
239                                CoverageItemKind::Branch {
240                                    branch_id,
241                                    path_id: 0,
242                                    is_first_opcode: true,
243                                },
244                                &true_body.src,
245                            );
246                            // Process the true body.
247                            self.visit_block_or_statement(&true_body)?;
248                        }
249                    }
250                }
251
252                Ok(())
253            }
254            NodeType::YulIf => {
255                self.visit_expression(
256                    &node
257                        .attribute("condition")
258                        .ok_or_else(|| eyre::eyre!("yul if statement had no condition"))?,
259                )?;
260                let body = node
261                    .body
262                    .as_deref()
263                    .ok_or_else(|| eyre::eyre!("yul if statement had no body"))?;
264
265                // We need to store the current branch ID here since visiting the body of either of
266                // the if blocks may increase `self.branch_id` in the case of nested if statements.
267                let branch_id = self.branch_id;
268
269                // We increase the branch ID here such that nested branches do not use the same
270                // branch ID as we do
271                self.branch_id += 1;
272
273                self.push_item_kind(
274                    CoverageItemKind::Branch { branch_id, path_id: 0, is_first_opcode: false },
275                    &node.src,
276                );
277                self.visit_block(body)?;
278
279                Ok(())
280            }
281            // Try-catch statement. Coverage is reported for expression, for each clause and their
282            // bodies (if any).
283            NodeType::TryStatement => {
284                self.visit_expression(
285                    &node
286                        .attribute("externalCall")
287                        .ok_or_else(|| eyre::eyre!("try statement had no call"))?,
288                )?;
289
290                // Add coverage for each Try-catch clause.
291                for clause in node
292                    .attribute::<Vec<Node>>("clauses")
293                    .ok_or_else(|| eyre::eyre!("try statement had no clause"))?
294                {
295                    // Add coverage for clause statement.
296                    self.push_item_kind(CoverageItemKind::Statement, &clause.src);
297                    self.visit_statement(&clause)?;
298
299                    // Add coverage for clause body only if it is not empty.
300                    if let Some(block) = clause.attribute::<Node>("block") {
301                        if has_statements(&block) {
302                            self.push_item_kind(CoverageItemKind::Statement, &block.src);
303                            self.visit_block(&block)?;
304                        }
305                    }
306                }
307
308                Ok(())
309            }
310            NodeType::YulSwitch => {
311                // Add coverage for each case statement amd their bodies.
312                for case in node
313                    .attribute::<Vec<Node>>("cases")
314                    .ok_or_else(|| eyre::eyre!("yul switch had no case"))?
315                {
316                    self.push_item_kind(CoverageItemKind::Statement, &case.src);
317                    self.visit_statement(&case)?;
318
319                    if let Some(body) = case.body {
320                        self.push_item_kind(CoverageItemKind::Statement, &body.src);
321                        self.visit_block(&body)?
322                    }
323                }
324                Ok(())
325            }
326            NodeType::YulForLoop => {
327                if let Some(condition) = node.attribute("condition") {
328                    self.visit_expression(&condition)?;
329                }
330                if let Some(pre) = node.attribute::<Node>("pre") {
331                    self.visit_block(&pre)?
332                }
333                if let Some(post) = node.attribute::<Node>("post") {
334                    self.visit_block(&post)?
335                }
336
337                if let Some(body) = &node.body {
338                    self.push_item_kind(CoverageItemKind::Statement, &body.src);
339                    self.visit_block(body)?
340                }
341                Ok(())
342            }
343            NodeType::YulFunctionDefinition => self.visit_modifier_or_yul_fn_definition(node),
344            _ => {
345                warn!("unexpected node type, expected a statement: {:?}", node.node_type);
346                Ok(())
347            }
348        }
349    }

351    fn visit_expression(&mut self, node: &Node) -> eyre::Result<()> {
352        match node.node_type {
353            NodeType::Assignment |
354            NodeType::UnaryOperation |
355            NodeType::Conditional |
356            NodeType::YulFunctionCall => {
357                self.push_item_kind(CoverageItemKind::Statement, &node.src);
358                Ok(())
359            }
360            NodeType::FunctionCall => {
361                // Do not count other kinds of calls towards coverage (like `typeConversion`
362                // and `structConstructorCall`).
363                let kind: Option<String> = node.attribute("kind");
364                if let Some("functionCall") = kind.as_deref() {
365                    self.push_item_kind(CoverageItemKind::Statement, &node.src);
366
367                    let expr: Option<Node> = node.attribute("expression");
368                    if let Some(NodeType::Identifier) = expr.as_ref().map(|expr| &expr.node_type) {
369                        // Might be a require call, add branch coverage.
370                        let name: Option<String> = expr.and_then(|expr| expr.attribute("name"));
371                        if let Some("require" | "assert") = name.as_deref() {
372                            let branch_id = self.branch_id;
373                            self.branch_id += 1;
374                            self.push_item_kind(
375                                CoverageItemKind::Branch {
376                                    branch_id,
377                                    path_id: 0,
378                                    is_first_opcode: false,
379                                },
380                                &node.src,
381                            );
382                            self.push_item_kind(
383                                CoverageItemKind::Branch {
384                                    branch_id,
385                                    path_id: 1,
386                                    is_first_opcode: false,
387                                },
388                                &node.src,
389                            );
390                        }
391                    }
392                }
393
394                Ok(())
395            }
396            NodeType::BinaryOperation => {
397                self.push_item_kind(CoverageItemKind::Statement, &node.src);
398
399                // visit left and right expressions
400                // There could possibly a function call in the left or right expression
401                // e.g: callFunc(a) + callFunc(b)
402                if let Some(expr) = node.attribute("leftExpression") {
403                    self.visit_expression(&expr)?;
404                }
405
406                if let Some(expr) = node.attribute("rightExpression") {
407                    self.visit_expression(&expr)?;
408                }
409
410                Ok(())
411            }
412            // Does not count towards coverage
413            NodeType::FunctionCallOptions |
414            NodeType::Identifier |
415            NodeType::IndexAccess |
416            NodeType::IndexRangeAccess |
417            NodeType::Literal |
418            NodeType::YulLiteralValue |
419            NodeType::YulIdentifier => Ok(()),
420            _ => {
421                warn!("unexpected node type, expected an expression: {:?}", node.node_type);
422                Ok(())
423            }
424        }
425    }

427    fn visit_block_or_statement(&mut self, node: &Node) -> eyre::Result<()> {
428        match node.node_type {
429            NodeType::Block => self.visit_block(node),
430            NodeType::Break |
431            NodeType::Continue |
432            NodeType::DoWhileStatement |
433            NodeType::EmitStatement |
434            NodeType::ExpressionStatement |
435            NodeType::ForStatement |
436            NodeType::IfStatement |
437            NodeType::InlineAssembly |
438            NodeType::Return |
439            NodeType::RevertStatement |
440            NodeType::TryStatement |
441            NodeType::VariableDeclarationStatement |
442            NodeType::YulVariableDeclaration |
443            NodeType::WhileStatement => self.visit_statement(node),
444            // Skip placeholder statements as they are never referenced in source maps.
445            NodeType::PlaceholderStatement => Ok(()),
446            _ => {
447                warn!("unexpected node type, expected block or statement: {:?}", node.node_type);
448                Ok(())
449            }
450        }
451    }

453    /// Creates a coverage item for a given kind and source location. Pushes item to the internal
454    /// collection (plus additional coverage line if item is a statement).
455    fn push_item_kind(&mut self, kind: CoverageItemKind, src: &ast::LowFidelitySourceLocation) {
456        let item = CoverageItem { kind, loc: self.source_location_for(src), hits: 0 };
457        // Push a line item if we haven't already
458        if matches!(item.kind, CoverageItemKind::Statement | CoverageItemKind::Branch { .. }) &&
459            self.last_line < item.loc.line
460        {
461            self.items.push(CoverageItem {
462                kind: CoverageItemKind::Line,
463                loc: item.loc.clone(),
464                hits: 0,
465            });
466            self.last_line = item.loc.line;
467        }
468
469        self.items.push(item);
470    }
471
472    fn source_location_for(&self, loc: &ast::LowFidelitySourceLocation) -> SourceLocation {
473        let loc_start =
474            self.source.char_indices().map(|(i, _)| i).nth(loc.start).unwrap_or_default();
475        SourceLocation {
476            source_id: self.source_id,
477            contract_name: self.contract_name.clone(),
478            start: loc.start as u32,
479            length: loc.length.map(|x| x as u32),
480            line: self.source[..loc_start].lines().count(),
481        }
482    }
483}

485/// Helper function to check if a given node is or contains any statement.
486fn has_statements(node: &Node) -> bool {
487    match node.node_type {
488        NodeType::DoWhileStatement |
489        NodeType::EmitStatement |
490        NodeType::ExpressionStatement |
491        NodeType::ForStatement |
492        NodeType::IfStatement |
493        NodeType::RevertStatement |
494        NodeType::TryStatement |
495        NodeType::VariableDeclarationStatement |
496        NodeType::WhileStatement => true,
497        _ => {
498            let statements: Vec<Node> = node.attribute("statements").unwrap_or_default();
499            !statements.is_empty()
500        }
501    }

504/// [`SourceAnalyzer`] result type.
505#[derive(Debug)]
506pub struct SourceAnalysis {
507    /// A collection of coverage items.
508    pub items: Vec<CoverageItem>,
509}
510
511/// Analyzes a set of sources to find coverage items.
512#[derive(Debug)]
513pub struct SourceAnalyzer<'a> {
514    sources: &'a SourceFiles<'a>,
515}
516
517impl<'a> SourceAnalyzer<'a> {
518    /// Creates a new source analyzer.
519    pub fn new(data: &'a SourceFiles<'a>) -> Self {
520        Self { sources: data }
521    }
522
523    /// Analyzes contracts in the sources held by the source analyzer.
524    ///
525    /// Coverage items are found by:
526    /// - Walking the AST of each contract (except interfaces)
527    /// - Recording the items of each contract
528    ///
529    /// Each coverage item contains relevant information to find opcodes corresponding to them: the
530    /// source ID the item is in, the source code range of the item, and the contract name the item
531    /// is in.
532    ///
533    /// Note: Source IDs are only unique per compilation job; that is, a code base compiled with
534    /// two different solc versions will produce overlapping source IDs if the compiler version is
535    /// not taken into account.
536    pub fn analyze(&self) -> eyre::Result<SourceAnalysis> {
537        let items = self
538            .sources
539            .sources
540            .par_iter()
541            .flat_map_iter(|(&source_id, SourceFile { source, ast })| {
542                ast.nodes.iter().map(move |node| {
543                    if !matches!(node.node_type, NodeType::ContractDefinition) {
544                        return Ok(vec![]);
545                    }
546
547                    // Skip interfaces which have no function implementations.
548                    let contract_kind: String = node
549                        .attribute("contractKind")
550                        .ok_or_else(|| eyre::eyre!("Contract has no kind"))?;
551                    if contract_kind == "interface" {
552                        return Ok(vec![]);
553                    }
554
555                    let name = node
556                        .attribute("name")
557                        .ok_or_else(|| eyre::eyre!("Contract has no name"))?;
558
559                    let mut visitor = ContractVisitor::new(source_id, source, &name);
560                    visitor.visit_contract(node)?;
561                    let mut items = visitor.items;
562
563                    let is_test = items.iter().any(|item| {
564                        if let CoverageItemKind::Function { name } = &item.kind {
565                            name.is_any_test()
566                        } else {
567                            false
568                        }
569                    });
570                    if is_test {
571                        items.clear();
572                    }
573
574                    Ok(items)
575                })
576            })
577            .collect::<eyre::Result<Vec<Vec<_>>>>()?;
578        Ok(SourceAnalysis { items: items.concat() })
579    }
580}
