18/// A simple summary reporter that prints the coverage results in a table.
19pub struct SummaryReporter {
20    /// The summary table.
21    table: Table,
22    /// The total coverage of the entire project.
23    total: CoverageSummary,
24}
25
26impl Default for SummaryReporter {
27    fn default() -> Self {
28        let mut table = Table::new();
29        table.load_preset(ASCII_MARKDOWN);
30        table.set_header(["File", "% Lines", "% Statements", "% Branches", "% Funcs"]);
31
32        Self { table, total: CoverageSummary::default() }
33    }
34}
35
36impl SummaryReporter {
37    fn add_row(&mut self, name: impl Into<Cell>, summary: CoverageSummary) {
38        let mut row = Row::new();
39        row.add_cell(name.into())
40            .add_cell(format_cell(summary.line_hits, summary.line_count))
41            .add_cell(format_cell(summary.statement_hits, summary.statement_count))
42            .add_cell(format_cell(summary.branch_hits, summary.branch_count))
43            .add_cell(format_cell(summary.function_hits, summary.function_count));
44        self.table.add_row(row);
45    }
46}
47
48impl CoverageReporter for SummaryReporter {
49    fn report(mut self, report: &CoverageReport) -> eyre::Result<()> {
50        for (path, summary) in report.summary_by_file() {
51            self.total += &summary;
52            self.add_row(path.display(), summary);
53        }
54
55        self.add_row("Total", self.total.clone());
56        println!("{}", self.table);
57        Ok(())
58    }
59}
60
61fn format_cell(hits: usize, total: usize) -> Cell {
62    let percentage = if total == 0 { 1. } else { hits as f64 / total as f64 };
63
64    let mut cell =
65        Cell::new(format!("{:.2}% ({hits}/{total})", percentage * 100.)).fg(match percentage {
66            _ if total == 0 => Color::Grey,
67            _ if percentage < 0.5 => Color::Red,
68            _ if percentage < 0.75 => Color::Yellow,
69            _ => Color::Green,
70        });
71
72    if total == 0 {
73        cell = cell.add_attribute(Attribute::Dim);
74    }
75    cell
76}

78pub struct LcovReporter<'a> {
79    /// Destination buffer
80    destination: &'a mut (dyn Write + 'a),
81}
82
83impl<'a> LcovReporter<'a> {
84    pub fn new(destination: &'a mut (dyn Write + 'a)) -> Self {
85        Self { destination }
86    }
87}
88
89impl<'a> CoverageReporter for LcovReporter<'a> {
90    fn report(self, report: &CoverageReport) -> eyre::Result<()> {
91        for (file, items) in report.items_by_source() {
92            let summary = items.iter().fold(CoverageSummary::default(), |mut summary, item| {
93                summary += item;
94                summary
95            });
96
97            writeln!(self.destination, "TN:")?;
98            writeln!(self.destination, "SF:{}", file.display())?;
99
100            for item in items {
101                let line = item.loc.line;
102                let hits = item.hits;
103                match item.kind {
104                    CoverageItemKind::Function { name } => {
105                        let name = format!("{}.{name}", item.loc.contract_name);
106                        writeln!(self.destination, "FN:{line},{name}")?;
107                        writeln!(self.destination, "FNDA:{hits},{name}")?;
108                    }
109                    CoverageItemKind::Line => {
110                        writeln!(self.destination, "DA:{line},{hits}")?;
111                    }
112                    CoverageItemKind::Branch { branch_id, path_id, .. } => {
113                        writeln!(
114                            self.destination,
115                            "BRDA:{line},{branch_id},{path_id},{}",
116                            if hits == 0 { "-".to_string() } else { hits.to_string() }
117                        )?;
118                    }
119                    // Statements are not in the LCOV format.
120                    // We don't add them in order to avoid doubling line hits.
121                    _ => {}
122                }
123            }
124
125            // Function summary
126            writeln!(self.destination, "FNF:{}", summary.function_count)?;
127            writeln!(self.destination, "FNH:{}", summary.function_hits)?;
128
129            // Line summary
130            writeln!(self.destination, "LF:{}", summary.line_count)?;
131            writeln!(self.destination, "LH:{}", summary.line_hits)?;
132
133            // Branch summary
134            writeln!(self.destination, "BRF:{}", summary.branch_count)?;
135            writeln!(self.destination, "BRH:{}", summary.branch_hits)?;
136
137            writeln!(self.destination, "end_of_record")?;
138        }
139
140        println!("Wrote LCOV report.");
141
142        Ok(())
143    }
144}

146/// A super verbose reporter for debugging coverage while it is still unstable.
147pub struct DebugReporter;
148
149impl CoverageReporter for DebugReporter {
150    fn report(self, report: &CoverageReport) -> eyre::Result<()> {
151        for (path, items) in report.items_by_source() {
152            println!("Uncovered for {}:", path.display());
153            items.iter().for_each(|item| {
154                if item.hits == 0 {
155                    println!("- {item}");
156                }
157            });
158            println!();
159        }
160
161        for (contract_id, anchors) in &report.anchors {
162            println!("Anchors for {contract_id}:");
163            anchors
164                .0
165                .iter()
166                .map(|anchor| (false, anchor))
167                .chain(anchors.1.iter().map(|anchor| (true, anchor)))
168                .for_each(|(is_deployed, anchor)| {
169                    println!("- {anchor}");
170                    if is_deployed {
171                        println!("- Creation code");
172                    } else {
173                        println!("- Runtime code");
174                    }
175                    println!(
176                        "  - Refers to item: {}",
177                        report
178                            .items
179                            .get(&contract_id.version)
180                            .and_then(|items| items.get(anchor.item_id))
181                            .map_or("None".to_owned(), |item| item.to_string())
182                    );
183                });
184            println!();
185        }
186
187        Ok(())
188    }
189}

191pub struct BytecodeReporter {
192    root: PathBuf,
193    destdir: PathBuf,
194}
195
196impl BytecodeReporter {
197    pub fn new(root: PathBuf, destdir: PathBuf) -> Self {
198        Self { root, destdir }
199    }
200}
201
202impl CoverageReporter for BytecodeReporter {
203    fn report(self, report: &CoverageReport) -> eyre::Result<()> {
204        use std::fmt::Write;
205
206        let no_source_elements = Vec::new();
207        let mut line_number_cache = LineNumberCache::new(self.root.clone());
208
209        for (contract_id, hits) in &report.bytecode_hits {
210            let ops = disassemble_bytes(hits.bytecode.to_vec())?;
211            let mut formatted = String::new();
212
213            let source_elements =
214                report.source_maps.get(contract_id).map(|sm| &sm.1).unwrap_or(&no_source_elements);
215
216            for (code, source_element) in std::iter::zip(ops.iter(), source_elements) {
217                let hits = hits
218                    .hits
219                    .get(&(code.offset as usize))
220                    .map(|h| format!("[{h:03}]"))
221                    .unwrap_or("     ".to_owned());
222                let source_id = source_element.index();
223                let source_path = source_id.and_then(|i| {
224                    report.source_paths.get(&(contract_id.version.clone(), i as usize))
225                });
226
227                let code = format!("{code:?}");
228                let start = source_element.offset() as usize;
229                let end = (source_element.offset() + source_element.length()) as usize;
230
231                if let Some(source_path) = source_path {
232                    let (sline, spos) = line_number_cache.get_position(source_path, start)?;
233                    let (eline, epos) = line_number_cache.get_position(source_path, end)?;
234                    writeln!(
235                        formatted,
236                        "{} {:40} // {}: {}:{}-{}:{} ({}-{})",
237                        hits,
238                        code,
239                        source_path.display(),
240                        sline,
241                        spos,
242                        eline,
243                        epos,
244                        start,
245                        end
246                    )?;
247                } else if let Some(source_id) = source_id {
248                    writeln!(formatted, "{hits} {code:40} // SRCID{source_id}: ({start}-{end})")?;
249                } else {
250                    writeln!(formatted, "{hits} {code:40}")?;
251                }
252            }
253            fs::write(
254                self.destdir.join(&*contract_id.contract_name).with_extension("asm"),
255                formatted,
256            )?;
257        }
258
259        Ok(())
260    }
261}

263/// Cache line number offsets for source files
264struct LineNumberCache {
265    root: PathBuf,
266    line_offsets: HashMap<PathBuf, Vec<usize>>,
267}
268
269impl LineNumberCache {
270    pub fn new(root: PathBuf) -> Self {
271        Self { root, line_offsets: HashMap::new() }
272    }
273
274    pub fn get_position(&mut self, path: &Path, offset: usize) -> eyre::Result<(usize, usize)> {
275        let line_offsets = match self.line_offsets.entry(path.to_path_buf()) {
276            hash_map::Entry::Occupied(o) => o.into_mut(),
277            hash_map::Entry::Vacant(v) => {
278                let text = fs::read_to_string(self.root.join(path))?;
279                let mut line_offsets = vec![0];
280                for line in text.lines() {
281                    let line_offset = line.as_ptr() as usize - text.as_ptr() as usize;
282                    line_offsets.push(line_offset);
283                }
284                v.insert(line_offsets)
285            }
286        };
287        let lo = match line_offsets.binary_search(&offset) {
288            Ok(lo) => lo,
289            Err(lo) => lo - 1,
290        };
291        let pos = offset - line_offsets.get(lo).unwrap() + 1;
292        Ok((lo, pos))
293    }
