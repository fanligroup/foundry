17/// Represents the gas report for a set of contracts.
18#[derive(Clone, Debug, Default, Serialize, Deserialize)]
19pub struct GasReport {
20    /// Whether to report any contracts.
21    report_any: bool,
22    /// Contracts to generate the report for.
23    report_for: HashSet<String>,
24    /// Contracts to ignore when generating the report.
25    ignore: HashSet<String>,
26    /// All contracts that were analyzed grouped by their identifier
27    /// ``test/Counter.t.sol:CounterTest
28    pub contracts: BTreeMap<String, ContractInfo>,
29}
30
31impl GasReport {
32    pub fn new(
33        report_for: impl IntoIterator<Item = String>,
34        ignore: impl IntoIterator<Item = String>,
35    ) -> Self {
36        let report_for = report_for.into_iter().collect::<HashSet<_>>();
37        let ignore = ignore.into_iter().collect::<HashSet<_>>();
38        let report_any = report_for.is_empty() || report_for.contains("*");
39        Self { report_any, report_for, ignore, ..Default::default() }
40    }
41
42    /// Whether the given contract should be reported.
43    #[instrument(level = "trace", skip(self), ret)]
44    fn should_report(&self, contract_name: &str) -> bool {
45        if self.ignore.contains(contract_name) {
46            let contains_anyway = self.report_for.contains(contract_name);
47            if contains_anyway {
48                // If the user listed the contract in 'gas_reports' (the foundry.toml field) a
49                // report for the contract is generated even if it's listed in the ignore
50                // list. This is addressed this way because getting a report you don't expect is
51                // preferable than not getting one you expect. A warning is printed to stderr
52                // indicating the "double listing".
53                eprintln!(
54                    "{}: {} is listed in both 'gas_reports' and 'gas_reports_ignore'.",
55                    "warning".yellow().bold(),
56                    contract_name
57                );
58            }
59            return contains_anyway;
60        }
61        self.report_any || self.report_for.contains(contract_name)
62    }
63
64    /// Analyzes the given traces and generates a gas report.
65    pub async fn analyze(
66        &mut self,
67        arenas: impl IntoIterator<Item = &CallTraceArena>,
68        decoder: &CallTraceDecoder,
69    ) {
70        for node in arenas.into_iter().flat_map(|arena| arena.nodes()) {
71            self.analyze_node(node, decoder).await;
72        }
73    }
74
75    async fn analyze_node(&mut self, node: &CallTraceNode, decoder: &CallTraceDecoder) {
76        let trace = &node.trace;
77
78        if trace.address == CHEATCODE_ADDRESS || trace.address == HARDHAT_CONSOLE_ADDRESS {
79            return;
80        }
81
82        // Only include top-level calls which account for calldata and base (21.000) cost.
83        // Only include Calls and Creates as only these calls are isolated in inspector.
84        if trace.depth > 1 &&
85            (trace.kind == CallKind::Call ||
86                trace.kind == CallKind::Create ||
87                trace.kind == CallKind::Create2 ||
88                trace.kind == CallKind::EOFCreate)
89        {
90            return;
91        }
92
93        let Some(name) = decoder.contracts.get(&node.trace.address) else { return };
94        let contract_name = name.rsplit(':').next().unwrap_or(name);
95
96        if !self.should_report(contract_name) {
97            return;
98        }
99
100        let decoded = || decoder.decode_function(&node.trace);
101
102        let contract_info = self.contracts.entry(name.to_string()).or_default();
103        if trace.kind.is_any_create() {
104            trace!(contract_name, "adding create gas info");
105            contract_info.gas = trace.gas_used;
106            contract_info.size = trace.data.len();
107        } else if let Some(DecodedCallData { signature, .. }) = decoded().await.call_data {
108            let name = signature.split('(').next().unwrap();
109            // ignore any test/setup functions
110            if !name.test_function_kind().is_known() {
111                trace!(contract_name, signature, "adding gas info");
112                let gas_info = contract_info
113                    .functions
114                    .entry(name.to_string())
115                    .or_default()
116                    .entry(signature.clone())
117                    .or_default();
118                gas_info.calls.push(trace.gas_used);
119            }
120        }
121    }
122
123    /// Finalizes the gas report by calculating the min, max, mean, and median for each function.
124    #[must_use]
125    pub fn finalize(mut self) -> Self {
126        trace!("finalizing gas report");
127        for contract in self.contracts.values_mut() {
128            for sigs in contract.functions.values_mut() {
129                for func in sigs.values_mut() {
130                    func.calls.sort_unstable();
131                    func.min = func.calls.first().copied().unwrap_or_default();
132                    func.max = func.calls.last().copied().unwrap_or_default();
133                    func.mean = calc::mean(&func.calls);
134                    func.median = calc::median_sorted(&func.calls);
135                }
136            }
137        }
138        self
139    }
140}

142impl Display for GasReport {
143    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> Result<(), std::fmt::Error> {
144        for (name, contract) in &self.contracts {
145            if contract.functions.is_empty() {
146                trace!(name, "gas report contract without functions");
147                continue;
148            }
149
150            let mut table = Table::new();
151            table.load_preset(ASCII_MARKDOWN);
152            table.set_header([Cell::new(format!("{name} contract"))
153                .add_attribute(Attribute::Bold)
154                .fg(Color::Green)]);
155            table.add_row([
156                Cell::new("Deployment Cost").add_attribute(Attribute::Bold).fg(Color::Cyan),
157                Cell::new("Deployment Size").add_attribute(Attribute::Bold).fg(Color::Cyan),
158            ]);
159            table.add_row([contract.gas.to_string(), contract.size.to_string()]);
160
161            table.add_row([
162                Cell::new("Function Name").add_attribute(Attribute::Bold).fg(Color::Magenta),
163                Cell::new("min").add_attribute(Attribute::Bold).fg(Color::Green),
164                Cell::new("avg").add_attribute(Attribute::Bold).fg(Color::Yellow),
165                Cell::new("median").add_attribute(Attribute::Bold).fg(Color::Yellow),
166                Cell::new("max").add_attribute(Attribute::Bold).fg(Color::Red),
167                Cell::new("# calls").add_attribute(Attribute::Bold),
168            ]);
169            contract.functions.iter().for_each(|(fname, sigs)| {
170                sigs.iter().for_each(|(sig, gas_info)| {
171                    // show function signature if overloaded else name
172                    let fn_display =
173                        if sigs.len() == 1 { fname.clone() } else { sig.replace(':', "") };
174
175                    table.add_row([
176                        Cell::new(fn_display).add_attribute(Attribute::Bold),
177                        Cell::new(gas_info.min.to_string()).fg(Color::Green),
178                        Cell::new(gas_info.mean.to_string()).fg(Color::Yellow),
179                        Cell::new(gas_info.median.to_string()).fg(Color::Yellow),
180                        Cell::new(gas_info.max.to_string()).fg(Color::Red),
181                        Cell::new(gas_info.calls.len().to_string()),
182                    ]);
183                })
184            });
185            writeln!(f, "{table}")?;
186            writeln!(f, "\n")?;
187        }
188        Ok(())
189    }
190}
