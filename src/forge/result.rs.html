25/// The aggregated result of a test run.
26#[derive(Clone, Debug)]
27pub struct TestOutcome {
28    /// The results of all test suites by their identifier (`path:contract_name`).
29    ///
30    /// Essentially `identifier => signature => result`.
31    pub results: BTreeMap<String, SuiteResult>,
32    /// Whether to allow test failures without failing the entire test run.
33    pub allow_failure: bool,
34    /// The decoder used to decode traces and logs.
35    ///
36    /// This is `None` if traces and logs were not decoded.
37    ///
38    /// Note that `Address` fields only contain the last executed test case's data.
39    pub last_run_decoder: Option<CallTraceDecoder>,
40    /// The gas report, if requested.
41    pub gas_report: Option<GasReport>,
42}
43
44impl TestOutcome {
45    /// Creates a new test outcome with the given results.
46    pub fn new(results: BTreeMap<String, SuiteResult>, allow_failure: bool) -> Self {
47        Self { results, allow_failure, last_run_decoder: None, gas_report: None }
48    }
49
50    /// Creates a new empty test outcome.
51    pub fn empty(allow_failure: bool) -> Self {
52        Self::new(BTreeMap::new(), allow_failure)
53    }
54
55    /// Returns an iterator over all individual succeeding tests and their names.
56    pub fn successes(&self) -> impl Iterator<Item = (&String, &TestResult)> {
57        self.tests().filter(|(_, t)| t.status.is_success())
58    }
59
60    /// Returns an iterator over all individual skipped tests and their names.
61    pub fn skips(&self) -> impl Iterator<Item = (&String, &TestResult)> {
62        self.tests().filter(|(_, t)| t.status.is_skipped())
63    }
64
65    /// Returns an iterator over all individual failing tests and their names.
66    pub fn failures(&self) -> impl Iterator<Item = (&String, &TestResult)> {
67        self.tests().filter(|(_, t)| t.status.is_failure())
68    }
69
70    /// Returns an iterator over all individual tests and their names.
71    pub fn tests(&self) -> impl Iterator<Item = (&String, &TestResult)> {
72        self.results.values().flat_map(|suite| suite.tests())
73    }
74
75    /// Flattens the test outcome into a list of individual tests.
76    // TODO: Replace this with `tests` and make it return `TestRef<'_>`
77    pub fn into_tests_cloned(&self) -> impl Iterator<Item = SuiteTestResult> + '_ {
78        self.results
79            .iter()
80            .flat_map(|(file, suite)| {
81                suite
82                    .test_results
83                    .iter()
84                    .map(move |(sig, result)| (file.clone(), sig.clone(), result.clone()))
85            })
86            .map(|(artifact_id, signature, result)| SuiteTestResult {
87                artifact_id,
88                signature,
89                result,
90            })
91    }
92
93    /// Flattens the test outcome into a list of individual tests.
94    pub fn into_tests(self) -> impl Iterator<Item = SuiteTestResult> {
95        self.results
96            .into_iter()
97            .flat_map(|(file, suite)| {
98                suite.test_results.into_iter().map(move |t| (file.clone(), t))
99            })
100            .map(|(artifact_id, (signature, result))| SuiteTestResult {
101                artifact_id,
102                signature,
103                result,
104            })
105    }

107    /// Returns the number of tests that passed.
108    pub fn passed(&self) -> usize {
109        self.successes().count()
110    }
111
112    /// Returns the number of tests that were skipped.
113    pub fn skipped(&self) -> usize {
114        self.skips().count()
115    }
116
117    /// Returns the number of tests that failed.
118    pub fn failed(&self) -> usize {
119        self.failures().count()
120    }
121
122    /// Sums up all the durations of all individual test suites.
123    ///
124    /// Note that this is not necessarily the wall clock time of the entire test run.
125    pub fn total_time(&self) -> Duration {
126        self.results.values().map(|suite| suite.duration).sum()
127    }
128
129    /// Formats the aggregated summary of all test suites into a string (for printing).
130    pub fn summary(&self, wall_clock_time: Duration) -> String {
131        let num_test_suites = self.results.len();
132        let suites = if num_test_suites == 1 { "suite" } else { "suites" };
133        let total_passed = self.passed();
134        let total_failed = self.failed();
135        let total_skipped = self.skipped();
136        let total_tests = total_passed + total_failed + total_skipped;
137        format!(
138            "\nRan {} test {} in {:.2?} ({:.2?} CPU time): {} tests passed, {} failed, {} skipped ({} total tests)",
139            num_test_suites,
140            suites,
141            wall_clock_time,
142            self.total_time(),
143            total_passed.green(),
144            total_failed.red(),
145            total_skipped.yellow(),
146            total_tests
147        )
148    }
149
150    /// Checks if there are any failures and failures are disallowed.
151    pub fn ensure_ok(&self) -> eyre::Result<()> {
152        let outcome = self;
153        let failures = outcome.failures().count();
154        if outcome.allow_failure || failures == 0 {
155            return Ok(());
156        }
157
158        if !shell::verbosity().is_normal() {
159            // TODO: Avoid process::exit
160            std::process::exit(1);
161        }
162
163        shell::println("")?;
164        shell::println("Failing tests:")?;
165        for (suite_name, suite) in outcome.results.iter() {
166            let failed = suite.failed();
167            if failed == 0 {
168                continue;
169            }
170
171            let term = if failed > 1 { "tests" } else { "test" };
172            shell::println(format!("Encountered {failed} failing {term} in {suite_name}"))?;
173            for (name, result) in suite.failures() {
174                shell::println(result.short_result(name))?;
175            }
176            shell::println("")?;
177        }
178        let successes = outcome.passed();
179        shell::println(format!(
180            "Encountered a total of {} failing tests, {} tests succeeded",
181            failures.to_string().red(),
182            successes.to_string().green()
183        ))?;
184
185        // TODO: Avoid process::exit
186        std::process::exit(1);
187    }
188
189    /// Removes first test result, if any.
190    pub fn remove_first(&mut self) -> Option<(String, String, TestResult)> {
191        self.results.iter_mut().find_map(|(suite_name, suite)| {
192            if let Some(test_name) = suite.test_results.keys().next().cloned() {
193                let result = suite.test_results.remove(&test_name).unwrap();
194                Some((suite_name.clone(), test_name, result))
195            } else {
196                None
197            }
198        })
199    }
200}

202/// A set of test results for a single test suite, which is all the tests in a single contract.
203#[derive(Clone, Debug, Serialize)]
204pub struct SuiteResult {
205    /// Wall clock time it took to execute all tests in this suite.
206    #[serde(with = "humantime_serde")]
207    pub duration: Duration,
208    /// Individual test results: `test fn signature -> TestResult`.
209    pub test_results: BTreeMap<String, TestResult>,
210    /// Generated warnings.
211    pub warnings: Vec<String>,
212}
213
214impl SuiteResult {
215    pub fn new(
216        duration: Duration,
217        test_results: BTreeMap<String, TestResult>,
218        mut warnings: Vec<String>,
219    ) -> Self {
220        // Add deprecated cheatcodes warning, if any of them used in current test suite.
221        let mut deprecated_cheatcodes = HashMap::new();
222        for test_result in test_results.values() {
223            deprecated_cheatcodes.extend(test_result.deprecated_cheatcodes.clone());
224        }
225        if !deprecated_cheatcodes.is_empty() {
226            let mut warning =
227                "the following cheatcode(s) are deprecated and will be removed in future versions:"
228                    .to_string();
229            for (cheatcode, reason) in deprecated_cheatcodes {
230                write!(warning, "\n  {cheatcode}").unwrap();
231                if let Some(reason) = reason {
232                    write!(warning, ": {reason}").unwrap();
233                }
234            }
235            warnings.push(warning);
236        }
237
238        Self { duration, test_results, warnings }
239    }

242    pub fn successes(&self) -> impl Iterator<Item = (&String, &TestResult)> {
243        self.tests().filter(|(_, t)| t.status.is_success())
244    }

249    }

253        self.tests().filter(|(_, t)| t.status.is_failure())
254    }

257    pub fn passed(&self) -> usize {
258        self.successes().count()
259    }
260
261    /// Returns the number of tests that were skipped.
262    pub fn skipped(&self) -> usize {
263        self.skips().count()
264    }
265
266    /// Returns the number of tests that failed.
267    pub fn failed(&self) -> usize {
268        self.failures().count()
269    }
270
271    /// Iterator over all tests and their names
272    pub fn tests(&self) -> impl Iterator<Item = (&String, &TestResult)> {
273        self.test_results.iter()
274    }
275
276    /// Whether this test suite is empty.
277    pub fn is_empty(&self) -> bool {
278        self.test_results.is_empty()
279    }
280
281    /// The number of tests in this test suite.
282    pub fn len(&self) -> usize {
283        self.test_results.len()
284    }
285
286    /// Sums up all the durations of all individual tests in this suite.
287    ///
288    /// Note that this is not necessarily the wall clock time of the entire test suite.
289    pub fn total_time(&self) -> Duration {
290        self.test_results.values().map(|result| result.duration).sum()
291    }
292
293    /// Returns the summary of a single test suite.
294    pub fn summary(&self) -> String {
295        let failed = self.failed();
296        let result = if failed == 0 { "ok".green() } else { "FAILED".red() };
297        format!(
298            "Suite result: {}. {} passed; {} failed; {} skipped; finished in {:.2?} ({:.2?} CPU time)",
299            result,
300            self.passed().green(),
301            failed.red(),
302            self.skipped().yellow(),
303            self.duration,
304            self.total_time(),
305        )
306    }

369/// The result of an executed test.
370#[derive(Clone, Debug, Default, Serialize, Deserialize)]
371pub struct TestResult {
372    /// The test status, indicating whether the test case succeeded, failed, or was marked as
373    /// skipped. This means that the transaction executed properly, the test was marked as
374    /// skipped with vm.skip(), or that there was a revert and that the test was expected to
375    /// fail (prefixed with `testFail`)
376    pub status: TestStatus,
377
378    /// If there was a revert, this field will be populated. Note that the test can
379    /// still be successful (i.e self.success == true) when it's expected to fail.
380    pub reason: Option<String>,
381
382    /// Minimal reproduction test case for failing test
383    pub counterexample: Option<CounterExample>,
384
385    /// Any captured & parsed as strings logs along the test's execution which should
386    /// be printed to the user.
387    pub logs: Vec<Log>,
388
389    /// What kind of test this was
390    pub kind: TestKind,
391
392    /// Traces
393    #[serde(skip)]
394    pub traces: Traces,
395
396    /// Additional traces to use for gas report.
397    #[serde(skip)]
398    pub gas_report_traces: Vec<Vec<CallTraceArena>>,
399
400    /// Raw coverage info
401    #[serde(skip)]
402    pub coverage: Option<HitMaps>,
403
404    /// Labeled addresses
405    pub labeled_addresses: HashMap<Address, String>,
406
407    pub duration: Duration,
408
409    /// pc breakpoint char map
410    pub breakpoints: Breakpoints,
411
412    /// Deprecated cheatcodes (mapped to their replacements, if any) used in current test.
413    #[serde(skip)]
414    pub deprecated_cheatcodes: HashMap<&'static str, Option<&'static str>>,
415}
416
417impl fmt::Display for TestResult {
418    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
419        match self.status {
420            TestStatus::Success => "[PASS]".green().fmt(f),
421            TestStatus::Skipped => {
422                let mut s = String::from("[SKIP");
423                if let Some(reason) = &self.reason {
424                    write!(s, ": {reason}").unwrap();
425                }
426                s.push(']');
427                s.yellow().fmt(f)
428            }
429            TestStatus::Failure => {
430                let mut s = String::from("[FAIL");
431                if self.reason.is_some() || self.counterexample.is_some() {
432                    if let Some(reason) = &self.reason {
433                        write!(s, ": {reason}").unwrap();
434                    }
435
436                    if let Some(counterexample) = &self.counterexample {
437                        match counterexample {
438                            CounterExample::Single(ex) => {
439                                write!(s, "; counterexample: {ex}]").unwrap();
440                            }
441                            CounterExample::Sequence(sequence) => {
442                                s.push_str("]\n\t[Sequence]\n");
443                                for ex in sequence {
444                                    writeln!(s, "\t\t{ex}").unwrap();
445                                }
446                            }
447                        }
448                    } else {
449                        s.push(']');
450                    }
451                } else {
452                    s.push(']');
453                }
454                s.red().fmt(f)
455            }
456        }
457    }
458}

504    /// Returns the result for single test. Merges execution results (logs, labeled addresses,
505    /// traces and coverages) in initial setup results.
506    pub fn single_result(
507        mut self,
508        success: bool,
509        reason: Option<String>,
510        raw_call_result: RawCallResult,
511    ) -> Self {
512        self.kind =
513            TestKind::Unit { gas: raw_call_result.gas_used.wrapping_sub(raw_call_result.stipend) };
514
515        // Record logs, labels, traces and merge coverages.
516        self.logs.extend(raw_call_result.logs);
517        self.labeled_addresses.extend(raw_call_result.labels);
518        self.traces.extend(raw_call_result.traces.map(|traces| (TraceKind::Execution, traces)));
519        self.merge_coverages(raw_call_result.coverage);
520
521        self.status = match success {
522            true => TestStatus::Success,
523            false => TestStatus::Failure,
524        };
525        self.reason = reason;
526        self.duration = Duration::default();
527        self.gas_report_traces = Vec::new();
528
529        if let Some(cheatcodes) = raw_call_result.cheatcodes {
530            self.breakpoints = cheatcodes.breakpoints;
531            self.deprecated_cheatcodes = cheatcodes.deprecated;
532        }
533
534        self
535    }

537    /// Returns the result for a fuzzed test. Merges fuzz execution results (logs, labeled
538    /// addresses, traces and coverages) in initial setup results.
539    pub fn fuzz_result(mut self, result: FuzzTestResult) -> Self {
540        self.kind = TestKind::Fuzz {
541            median_gas: result.median_gas(false),
542            mean_gas: result.mean_gas(false),
543            first_case: result.first_case,
544            runs: result.gas_by_case.len(),
545        };
546
547        // Record logs, labels, traces and merge coverages.
548        self.logs.extend(result.logs);
549        self.labeled_addresses.extend(result.labeled_addresses);
550        self.traces.extend(result.traces.map(|traces| (TraceKind::Execution, traces)));
551        self.merge_coverages(result.coverage);
552
553        self.status = if result.skipped {
554            TestStatus::Skipped
555        } else if result.success {
556            TestStatus::Success
557        } else {
558            TestStatus::Failure
559        };
560        self.reason = result.reason;
561        self.counterexample = result.counterexample;
562        self.duration = Duration::default();
563        self.gas_report_traces = result.gas_report_traces.into_iter().map(|t| vec![t]).collect();
564        self.breakpoints = result.breakpoints.unwrap_or_default();
565        self.deprecated_cheatcodes = result.deprecated_cheatcodes;
566
567        self
568    }

604    /// Returns the invariant test result.
605    pub fn invariant_result(
606        mut self,
607        gas_report_traces: Vec<Vec<CallTraceArena>>,
608        success: bool,
609        reason: Option<String>,

616            calls: cases.iter().map(|sequence| sequence.cases().len()).sum(),

620            true => TestStatus::Success,
621            false => TestStatus::Failure,

639    /// Function to merge logs, addresses, traces and coverage from a call result into test result.
640    pub fn merge_call_result(&mut self, call_result: &RawCallResult) {
641        self.logs.extend(call_result.logs.clone());
642        self.labeled_addresses.extend(call_result.labels.clone());
643        self.traces.extend(call_result.traces.clone().map(|traces| (TraceKind::Execution, traces)));
644        self.merge_coverages(call_result.coverage.clone());
645    }
646
647    /// Function to merge given coverage in current test result coverage.
648    pub fn merge_coverages(&mut self, other_coverage: Option<HitMaps>) {
649        let old_coverage = std::mem::take(&mut self.coverage);
650        self.coverage = match (old_coverage, other_coverage) {
651            (Some(old_coverage), Some(other)) => Some(old_coverage.merged(other)),
652            (a, b) => a.or(b),
653        };
654    }

657/// Data report by a test.
658#[derive(Clone, Debug, PartialEq, Eq)]
659pub enum TestKindReport {
660    Unit { gas: u64 },
661    Fuzz { runs: usize, mean_gas: u64, median_gas: u64 },
662    Invariant { runs: usize, calls: usize, reverts: usize },
663}
664
665impl fmt::Display for TestKindReport {
666    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
667        match *self {
668            Self::Unit { gas } => {
669                write!(f, "(gas: {gas})")
670            }
671            Self::Fuzz { runs, mean_gas, median_gas } => {
672                write!(f, "(runs: {runs}, μ: {mean_gas}, ~: {median_gas})")
673            }
674            Self::Invariant { runs, calls, reverts } => {
675                write!(f, "(runs: {runs}, calls: {calls}, reverts: {reverts})")
676            }
677        }
678    }
679}

681impl TestKindReport {
682    /// Returns the main gas value to compare against
683    pub fn gas(&self) -> u64 {
684        match *self {
685            Self::Unit { gas } => gas,
686            // We use the median for comparisons
687            Self::Fuzz { median_gas, .. } => median_gas,
688            // We return 0 since it's not applicable
689            Self::Invariant { .. } => 0,
690        }
691    }

694/// Various types of tests
695#[derive(Clone, Debug, Serialize, Deserialize)]
696pub enum TestKind {
697    /// A unit test.
698    Unit { gas: u64 },
699    /// A fuzz test.
700    Fuzz {
701        /// we keep this for the debugger
702        first_case: FuzzCase,
703        runs: usize,
704        mean_gas: u64,
705        median_gas: u64,
706    },
707    /// An invariant test.
708    Invariant { runs: usize, calls: usize, reverts: usize },
709}
710
711impl Default for TestKind {
712    fn default() -> Self {
713        Self::Unit { gas: 0 }
714    }
715}
716
717impl TestKind {
718    /// The gas consumed by this test
719    pub fn report(&self) -> TestKindReport {
720        match *self {
721            Self::Unit { gas } => TestKindReport::Unit { gas },
722            Self::Fuzz { first_case: _, runs, mean_gas, median_gas } => {
723                TestKindReport::Fuzz { runs, mean_gas, median_gas }
724            }
725            Self::Invariant { runs, calls, reverts } => {
726                TestKindReport::Invariant { runs, calls, reverts }
727            }
728        }
729    }
730}

732#[derive(Clone, Debug, Default)]
733pub struct TestSetup {
734    /// The address at which the test contract was deployed
735    pub address: Address,
736    /// The logs emitted during setup
737    pub logs: Vec<Log>,
738    /// Call traces of the setup
739    pub traces: Traces,
740    /// Addresses labeled during setup
741    pub labeled_addresses: HashMap<Address, String>,
742    /// The reason the setup failed, if it did
743    pub reason: Option<String>,
744    /// Coverage info during setup
745    pub coverage: Option<HitMaps>,
746    /// Defined fuzz test fixtures
747    pub fuzz_fixtures: FuzzFixtures,
748}
749
750impl TestSetup {
751    pub fn from_evm_error_with(
752        error: EvmError,
753        mut logs: Vec<Log>,
754        mut traces: Traces,
755        mut labeled_addresses: HashMap<Address, String>,
756    ) -> Self {
757        match error {
758            EvmError::Execution(err) => {
759                // force the tracekind to be setup so a trace is shown.
760                traces.extend(err.raw.traces.map(|traces| (TraceKind::Setup, traces)));
761                logs.extend(err.raw.logs);
762                labeled_addresses.extend(err.raw.labels);
763                Self::failed_with(logs, traces, labeled_addresses, err.reason)
764            }
765            e => Self::failed_with(
766                logs,
767                traces,
768                labeled_addresses,
769                format!("failed to deploy contract: {e}"),
770            ),
771        }
772    }
773
774    pub fn success(
775        address: Address,
776        logs: Vec<Log>,
777        traces: Traces,
778        labeled_addresses: HashMap<Address, String>,
779        coverage: Option<HitMaps>,
780        fuzz_fixtures: FuzzFixtures,
781    ) -> Self {
782        Self { address, logs, traces, labeled_addresses, reason: None, coverage, fuzz_fixtures }
783    }
784
785    pub fn failed_with(
786        logs: Vec<Log>,
787        traces: Traces,
788        labeled_addresses: HashMap<Address, String>,
789        reason: String,
790    ) -> Self {
791        Self {
792            address: Address::ZERO,
793            logs,
794            traces,
795            labeled_addresses,
796            reason: Some(reason),
797            coverage: None,
798            fuzz_fixtures: FuzzFixtures::default(),
799        }
800    }
801
802    pub fn failed(reason: String) -> Self {
803        Self { reason: Some(reason), ..Default::default() }
804    }
805}
