157        let Self { privateKey } = self;

159            parse_private_key_p256(privateKey)?.verifying_key().as_affine().to_encoded_point(false);

164    }

167/// Using a given private key, return its public ETH address, its public key affine x and y
168/// coordinates, and its private key (see the 'Wallet' struct)
169///
170/// If 'label' is set to 'Some()', assign that label to the associated ETH address in state
171fn create_wallet(private_key: &U256, label: Option<&str>, state: &mut Cheatcodes) -> Result {
172    let key = parse_private_key(private_key)?;
173    let addr = alloy_signer::utils::secret_key_to_address(&key);
174
175    let pub_key = key.verifying_key().as_affine().to_encoded_point(false);
176    let pub_key_x = U256::from_be_bytes((*pub_key.x().unwrap()).into());
177    let pub_key_y = U256::from_be_bytes((*pub_key.y().unwrap()).into());
178
179    if let Some(label) = label {
180        state.labels.insert(addr, label.into());
181    }
182
183    Ok(Wallet { addr, publicKeyX: pub_key_x, publicKeyY: pub_key_y, privateKey: *private_key }
184        .abi_encode())
185}

187fn encode_full_sig(sig: alloy_primitives::Signature) -> Vec<u8> {
188    // Retrieve v, r and s from signature.
189    let v = U256::from(sig.v().y_parity_byte_non_eip155().unwrap_or(sig.v().y_parity_byte()));
190    let r = B256::from(sig.r());
191    let s = B256::from(sig.s());
192    (v, r, s).abi_encode()
193}
194
195fn encode_compact_sig(sig: alloy_primitives::Signature) -> Vec<u8> {
196    // Implement EIP-2098 compact signature.
197    let r = B256::from(sig.r());
198    let mut vs = sig.s();
199    vs.set_bit(255, sig.v().y_parity());
200    (r, vs).abi_encode()
201}

203fn sign(private_key: &U256, digest: &B256) -> Result<alloy_primitives::Signature> {
204    // The `ecrecover` precompile does not use EIP-155. No chain ID is needed.
205    let wallet = parse_wallet(private_key)?;
206    let sig = wallet.sign_hash_sync(digest)?;
207    debug_assert_eq!(sig.recover_address_from_prehash(digest)?, wallet.address());
208    Ok(sig)
209}

211fn sign_with_wallet(
212    state: &mut Cheatcodes,
213    signer: Option<Address>,
214    digest: &B256,
215) -> Result<alloy_primitives::Signature> {
216    let Some(script_wallets) = state.script_wallets() else {
217        bail!("no wallets are available");
218    };
219
220    let mut script_wallets = script_wallets.inner.lock();
221    let maybe_provided_sender = script_wallets.provided_sender;
222    let signers = script_wallets.multi_wallet.signers()?;
223
224    let signer = if let Some(signer) = signer {
225        signer
226    } else if let Some(provided_sender) = maybe_provided_sender {
227        provided_sender
228    } else if signers.len() == 1 {
229        *signers.keys().next().unwrap()
230    } else {
231        bail!("could not determine signer");
232    };
233
234    let wallet = signers
235        .get(&signer)
236        .ok_or_else(|| fmt_err!("signer with address {signer} is not available"))?;
237
238    let sig = foundry_common::block_on(wallet.sign_hash(digest))?;
239    debug_assert_eq!(sig.recover_address_from_prehash(digest)?, signer);
240    Ok(sig)
241}

243fn sign_p256(private_key: &U256, digest: &B256) -> Result {
244    let signing_key = parse_private_key_p256(private_key)?;
245    let signature: Signature = signing_key.sign_prehash(digest.as_slice())?;
246    let r_bytes: [u8; 32] = signature.r().to_bytes().into();
247    let s_bytes: [u8; 32] = signature.s().to_bytes().into();
248
249    Ok((r_bytes, s_bytes).abi_encode())
250}

252fn validate_private_key<C: ecdsa::PrimeCurve>(private_key: &U256) -> Result<()> {
253    ensure!(*private_key != U256::ZERO, "private key cannot be 0");
254    let order = U256::from_be_slice(&C::ORDER.to_be_byte_array());
255    ensure!(
256        *private_key < U256::from_be_slice(&C::ORDER.to_be_byte_array()),
257        "private key must be less than the {curve:?} curve order ({order})",
258        curve = C::default(),
259    );
260
261    Ok(())
262}

264fn parse_private_key(private_key: &U256) -> Result<SigningKey> {
265    validate_private_key::<k256::Secp256k1>(private_key)?;
266    Ok(SigningKey::from_bytes((&private_key.to_be_bytes()).into())?)
267}

269fn parse_private_key_p256(private_key: &U256) -> Result<P256SigningKey> {
270    validate_private_key::<p256::NistP256>(private_key)?;
271    Ok(P256SigningKey::from_bytes((&private_key.to_be_bytes()).into())?)
272}

278fn derive_key_str(mnemonic: &str, path: &str, index: u32, language: &str) -> Result {
279    match language {
280        "chinese_simplified" => derive_key::<ChineseSimplified>(mnemonic, path, index),
281        "chinese_traditional" => derive_key::<ChineseTraditional>(mnemonic, path, index),
282        "czech" => derive_key::<Czech>(mnemonic, path, index),
283        "english" => derive_key::<English>(mnemonic, path, index),
284        "french" => derive_key::<French>(mnemonic, path, index),
285        "italian" => derive_key::<Italian>(mnemonic, path, index),
286        "japanese" => derive_key::<Japanese>(mnemonic, path, index),
287        "korean" => derive_key::<Korean>(mnemonic, path, index),
288        "portuguese" => derive_key::<Portuguese>(mnemonic, path, index),
289        "spanish" => derive_key::<Spanish>(mnemonic, path, index),
290        _ => Err(fmt_err!("unsupported mnemonic language: {language:?}")),
291    }
292}

294fn derive_key<W: Wordlist>(mnemonic: &str, path: &str, index: u32) -> Result {
295    fn derive_key_path(path: &str, index: u32) -> String {
296        let mut out = path.to_string();
297        if !out.ends_with('/') {
298            out.push('/');
299        }
300        out.push_str(&index.to_string());
301        out
302    }
303
304    let wallet = MnemonicBuilder::<W>::default()
305        .phrase(mnemonic)
306        .derivation_path(derive_key_path(path, index))?
307        .build()?;
308    let private_key = U256::from_be_bytes(wallet.credential().to_bytes().into());
309    Ok(private_key.abi_encode())
310}

320        use p256::ecdsa::VerifyingKey;

323        let signing_key = P256SigningKey::from_bytes(&pk_u256.to_be_bytes().into()).unwrap();
324        let digest = FixedBytes::from_hex(
325            "0x44acf6b7e36c1342c2c5897204fe09504e1e2efb1a900377dbc4e7a6a133ec56",
326        )
327        .unwrap();
328
329        let result = sign_p256(&pk_u256, &digest).unwrap();
330        let result_bytes: [u8; 64] = result.try_into().unwrap();
331        let signature = Signature::from_bytes(&result_bytes.into()).unwrap();
332        let verifying_key = VerifyingKey::from(&signing_key);
333        assert!(verifying_key.verify_prehash(digest.as_slice(), &signature).is_ok());
