30pub async fn estimate_gas<P, T>(
31    tx: &mut WithOtherFields<TransactionRequest>,
32    provider: &P,
33    estimate_multiplier: u64,
34) -> Result<()>
35where
36    P: Provider<T, AnyNetwork>,
37    T: Transport + Clone,
38{
39    // if already set, some RPC endpoints might simply return the gas value that is already
40    // set in the request and omit the estimate altogether, so we remove it here
41    tx.gas = None;
42
43    tx.set_gas_limit(
44        provider.estimate_gas(tx).await.wrap_err("Failed to estimate gas for tx")? *
45            estimate_multiplier as u128 /
46            100,
47    );
48    Ok(())
49}

57pub async fn send_transaction(
58    provider: Arc<RetryProvider>,
59    mut kind: SendTransactionKind<'_>,
60    sequential_broadcast: bool,
61    is_fixed_gas_limit: bool,
62    estimate_via_rpc: bool,
63    estimate_multiplier: u64,
64) -> Result<TxHash> {
65    if let SendTransactionKind::Raw(tx, _) | SendTransactionKind::Unlocked(tx) = &mut kind {
66        if sequential_broadcast {
67            let from = tx.from.expect("no sender");
68
69            let nonce = provider.get_transaction_count(from).await?;
70
71            let tx_nonce = tx.nonce.expect("no nonce");
72            if nonce != tx_nonce {
73                bail!("EOA nonce changed unexpectedly while sending transactions. Expected {tx_nonce} got {nonce} from provider.")
74            }
75        }
76
77        // Chains which use `eth_estimateGas` are being sent sequentially and require their
78        // gas to be re-estimated right before broadcasting.
79        if !is_fixed_gas_limit && estimate_via_rpc {
80            estimate_gas(tx, &provider, estimate_multiplier).await?;
81        }
82    }
83
84    let pending = match kind {
85        SendTransactionKind::Unlocked(tx) => {
86            debug!("sending transaction from unlocked account {:?}", tx);
87
88            // Submit the transaction
89            provider.send_transaction(tx).await?
90        }
91        SendTransactionKind::Raw(tx, signer) => {
92            debug!("sending transaction: {:?}", tx);
93            let signed = tx.build(signer).await?;
94
95            // Submit the raw transaction
96            provider.send_raw_transaction(signed.encoded_2718().as_ref()).await?
97        }
98        SendTransactionKind::Signed(tx) => {
99            debug!("sending transaction: {:?}", tx);
100            provider.send_raw_transaction(tx.encoded_2718().as_ref()).await?
101        }
102    };
103
104    Ok(*pending.tx_hash())
105}

107/// How to send a single transaction
108#[derive(Clone)]
109pub enum SendTransactionKind<'a> {
110    Unlocked(WithOtherFields<TransactionRequest>),
111    Raw(WithOtherFields<TransactionRequest>, &'a EthereumWallet),
112    Signed(TxEnvelope),
113}

115/// Represents how to send _all_ transactions
116pub enum SendTransactionsKind {
117    /// Send via `eth_sendTransaction` and rely on the  `from` address being unlocked.
118    Unlocked(HashSet<Address>),
119    /// Send a signed transaction via `eth_sendRawTransaction`
120    Raw(HashMap<Address, EthereumWallet>),
121}
122
123impl SendTransactionsKind {
124    /// Returns the [`SendTransactionKind`] for the given address
125    ///
126    /// Returns an error if no matching signer is found or the address is not unlocked
127    pub fn for_sender(
128        &self,
129        addr: &Address,
130        tx: WithOtherFields<TransactionRequest>,
131    ) -> Result<SendTransactionKind<'_>> {
132        match self {
133            Self::Unlocked(unlocked) => {
134                if !unlocked.contains(addr) {
135                    bail!("Sender address {:?} is not unlocked", addr)
136                }
137                Ok(SendTransactionKind::Unlocked(tx))
138            }
139            Self::Raw(wallets) => {
140                if let Some(wallet) = wallets.get(addr) {
141                    Ok(SendTransactionKind::Raw(tx, wallet))
142                } else {
143                    bail!("No matching signer for {:?} found", addr)
144                }
145            }
146        }
147    }
148}

162    pub async fn wait_for_pending(mut self) -> Result<Self> {
163        let progress = ScriptProgress::default();
164        let progress_ref = &progress;
165        let futs = self
166            .sequence
167            .sequences_mut()
168            .iter_mut()
169            .enumerate()
170            .map(|(sequence_idx, sequence)| async move {
171                let rpc_url = sequence.rpc_url();
172                let provider = Arc::new(get_http_provider(rpc_url));
173                progress_ref
174                    .wait_for_pending(
175                        sequence_idx,
176                        sequence,
177                        &provider,
178                        self.script_config.config.transaction_timeout,
179                    )
180                    .await
181            })
182            .collect::<Vec<_>>();
183
184        let errors = join_all(futs).await.into_iter().filter_map(Result::err).collect::<Vec<_>>();
185
186        self.sequence.save(true, false)?;
187
188        if !errors.is_empty() {
189            return Err(eyre::eyre!("{}", errors.iter().format("\n")));
190        }
191
192        Ok(self)
193    }

195    /// Broadcasts transactions from all sequences.
196    pub async fn broadcast(mut self) -> Result<BroadcastedState> {
197        let required_addresses = self
198            .sequence
199            .sequences()
200            .iter()
201            .flat_map(|sequence| sequence.transactions().map(|tx| tx.from().expect("missing from")))
202            .collect::<HashSet<_>>();
203
204        if required_addresses.contains(&Config::DEFAULT_SENDER) {
205            eyre::bail!(
206                "You seem to be using Foundry's default sender. Be sure to set your own --sender."
207            );
208        }
209
210        let send_kind = if self.args.unlocked {
211            SendTransactionsKind::Unlocked(required_addresses.clone())
212        } else {
213            let signers = self.script_wallets.into_multi_wallet().into_signers()?;
214            let mut missing_addresses = Vec::new();
215
216            for addr in &required_addresses {
217                if !signers.contains_key(addr) {
218                    missing_addresses.push(addr);
219                }
220            }
221
222            if !missing_addresses.is_empty() {
223                eyre::bail!(
224                    "No associated wallet for addresses: {:?}. Unlocked wallets: {:?}",
225                    missing_addresses,
226                    signers.keys().collect::<Vec<_>>()
227                );
228            }
229
230            let signers = signers
231                .into_iter()
232                .map(|(addr, signer)| (addr, EthereumWallet::new(signer)))
233                .collect();
234
235            SendTransactionsKind::Raw(signers)
236        };
237
238        let progress = ScriptProgress::default();
239
240        for i in 0..self.sequence.sequences().len() {
241            let mut sequence = self.sequence.sequences_mut().get_mut(i).unwrap();
242
243            let provider = Arc::new(try_get_http_provider(sequence.rpc_url())?);
244            let already_broadcasted = sequence.receipts.len();
245
246            let seq_progress = progress.get_sequence_progress(i, sequence);
247
248            if already_broadcasted < sequence.transactions.len() {
249                let is_legacy = Chain::from(sequence.chain).is_legacy() || self.args.legacy;
250                // Make a one-time gas price estimation
251                let (gas_price, eip1559_fees) = match (
252                    is_legacy,
253                    self.args.with_gas_price,
254                    self.args.priority_gas_price,
255                ) {
256                    (true, Some(gas_price), _) => (Some(gas_price.to()), None),
257                    (true, None, _) => (Some(provider.get_gas_price().await?), None),
258                    (false, Some(max_fee_per_gas), Some(max_priority_fee_per_gas)) => (
259                        None,
260                        Some(Eip1559Estimation {
261                            max_fee_per_gas: max_fee_per_gas.to(),
262                            max_priority_fee_per_gas: max_priority_fee_per_gas.to(),
263                        }),
264                    ),
265                    (false, _, _) => {
266                        let mut fees = provider.estimate_eip1559_fees(None).await.wrap_err("Failed to estimate EIP1559 fees. This chain might not support EIP1559, try adding --legacy to your command.")?;
267
268                        if let Some(gas_price) = self.args.with_gas_price {
269                            fees.max_fee_per_gas = gas_price.to();
270                        }
271
272                        if let Some(priority_gas_price) = self.args.priority_gas_price {
273                            fees.max_priority_fee_per_gas = priority_gas_price.to();
274                        }
275
276                        (None, Some(fees))
277                    }
278                };
279
280                // Iterate through transactions, matching the `from` field with the associated
281                // wallet. Then send the transaction. Panics if we find a unknown `from`
282                let transactions = sequence
283                    .transactions
284                    .iter()
285                    .skip(already_broadcasted)
286                    .map(|tx_with_metadata| {
287                        let is_fixed_gas_limit = tx_with_metadata.is_fixed_gas_limit;
288
289                        let kind = match tx_with_metadata.tx().clone() {
290                            TransactionMaybeSigned::Signed { tx, .. } => {
291                                SendTransactionKind::Signed(tx)
292                            }
293                            TransactionMaybeSigned::Unsigned(mut tx) => {
294                                let from = tx.from.expect("No sender for onchain transaction!");
295
296                                tx.set_chain_id(sequence.chain);
297
298                                // Set TxKind::Create explicitly to satisfy `check_reqd_fields` in
299                                // alloy
300                                if tx.to.is_none() {
301                                    tx.set_create();
302                                }
303
304                                if let Some(gas_price) = gas_price {
305                                    tx.set_gas_price(gas_price);
306                                } else {
307                                    let eip1559_fees = eip1559_fees.expect("was set above");
308                                    tx.set_max_priority_fee_per_gas(
309                                        eip1559_fees.max_priority_fee_per_gas,
310                                    );
311                                    tx.set_max_fee_per_gas(eip1559_fees.max_fee_per_gas);
312                                }
313
314                                send_kind.for_sender(&from, tx)?
315                            }
316                        };
317
318                        Ok((kind, is_fixed_gas_limit))
319                    })
320                    .collect::<Result<Vec<_>>>()?;
321
322                let estimate_via_rpc =
323                    has_different_gas_calc(sequence.chain) || self.args.skip_simulation;
324
325                // We only wait for a transaction receipt before sending the next transaction, if
326                // there is more than one signer. There would be no way of assuring
327                // their order otherwise.
328                // Or if the chain does not support batched transactions (eg. Arbitrum).
329                // Or if we need to invoke eth_estimateGas before sending transactions.
330                let sequential_broadcast = estimate_via_rpc ||
331                    self.args.slow ||
332                    required_addresses.len() != 1 ||
333                    !has_batch_support(sequence.chain);
334
335                // We send transactions and wait for receipts in batches.
336                let batch_size = if sequential_broadcast { 1 } else { self.args.batch_size };
337                let mut index = already_broadcasted;
338
339                for (batch_number, batch) in transactions.chunks(batch_size).enumerate() {
340                    let mut pending_transactions = vec![];
341
342                    seq_progress.inner.write().set_status(&format!(
343                        "Sending transactions [{} - {}]",
344                        batch_number * batch_size,
345                        batch_number * batch_size + std::cmp::min(batch_size, batch.len()) - 1
346                    ));
347                    for (kind, is_fixed_gas_limit) in batch {
348                        let fut = send_transaction(
349                            provider.clone(),
350                            kind.clone(),
351                            sequential_broadcast,
352                            *is_fixed_gas_limit,
353                            estimate_via_rpc,
354                            self.args.gas_estimate_multiplier,
355                        );
356                        pending_transactions.push(fut);
357                    }
358
359                    if !pending_transactions.is_empty() {
360                        let mut buffer = futures::stream::iter(pending_transactions).buffered(7);
361
362                        while let Some(tx_hash) = buffer.next().await {
363                            let tx_hash = tx_hash.wrap_err("Failed to send transaction")?;
364                            sequence.add_pending(index, tx_hash);
365
366                            // Checkpoint save
367                            self.sequence.save(true, false)?;
368                            sequence = self.sequence.sequences_mut().get_mut(i).unwrap();
369
370                            seq_progress.inner.write().tx_sent(tx_hash);
371                            index += 1;
372                        }
373
374                        // Checkpoint save
375                        self.sequence.save(true, false)?;
376                        sequence = self.sequence.sequences_mut().get_mut(i).unwrap();
377
378                        progress
379                            .wait_for_pending(
380                                i,
381                                sequence,
382                                &provider,
383                                self.script_config.config.transaction_timeout,
384                            )
385                            .await?
386                    }
387                    // Checkpoint save
388                    self.sequence.save(true, false)?;
389                    sequence = self.sequence.sequences_mut().get_mut(i).unwrap();
390                }
391            }
392
393            let (total_gas, total_gas_price, total_paid) =
394                sequence.receipts.iter().fold((0, 0, 0), |acc, receipt| {
395                    let gas_used = receipt.gas_used;
396                    let gas_price = receipt.effective_gas_price;
397                    (acc.0 + gas_used, acc.1 + gas_price, acc.2 + gas_used * gas_price)
398                });
399            let paid = format_units(total_paid, 18).unwrap_or_else(|_| "N/A".to_string());
400            let avg_gas_price = format_units(total_gas_price / sequence.receipts.len() as u128, 9)
401                .unwrap_or_else(|_| "N/A".to_string());
402
403            seq_progress.inner.write().set_status(&format!(
404                "Total Paid: {} ETH ({} gas * avg {} gwei)\n",
405                paid.trim_end_matches('0'),
406                total_gas,
407                avg_gas_price.trim_end_matches('0').trim_end_matches('.')
408            ));
409            seq_progress.inner.write().finish();
410        }
411
412        shell::println("\n\n==========================")?;
413        shell::println("\nONCHAIN EXECUTION COMPLETE & SUCCESSFUL.")?;
414
415        Ok(BroadcastedState {
416            args: self.args,
417            script_config: self.script_config,
418            build_data: self.build_data,
419            sequence: self.sequence,
420        })
421    }

423    pub fn verify_preflight_check(&self) -> Result<()> {
424        for sequence in self.sequence.sequences() {
425            if self.args.verifier.verifier == VerificationProviderType::Etherscan &&
426                self.script_config
427                    .config
428                    .get_etherscan_api_key(Some(sequence.chain.into()))
429                    .is_none()
430            {
431                eyre::bail!("Missing etherscan key for chain {}", sequence.chain);
432            }
433        }
434
435        Ok(())
436    }
