24impl DepositTransactionRequest {
25    pub fn hash(&self) -> B256 {
26        let mut encoded = Vec::new();
27        encoded.put_u8(DEPOSIT_TX_TYPE_ID);
28        self.encode(&mut encoded);
29
30        B256::from_slice(alloy_primitives::keccak256(encoded).as_slice())
31    }
32
33    /// Encodes only the transaction's fields into the desired buffer, without a RLP header.
34    pub(crate) fn encode_fields(&self, out: &mut dyn alloy_rlp::BufMut) {
35        self.source_hash.encode(out);
36        self.from.encode(out);
37        self.kind.encode(out);
38        self.mint.encode(out);
39        self.value.encode(out);
40        self.gas_limit.encode(out);
41        self.is_system_tx.encode(out);
42        self.input.encode(out);
43    }
44
45    /// Calculates the length of the RLP-encoded transaction's fields.
46    pub(crate) fn fields_len(&self) -> usize {
47        let mut len = 0;
48        len += self.source_hash.length();
49        len += self.from.length();
50        len += self.kind.length();
51        len += self.mint.length();
52        len += self.value.length();
53        len += self.gas_limit.length();
54        len += self.is_system_tx.length();
55        len += self.input.length();
56        len
57    }
58
59    /// Decodes the inner [DepositTransactionRequest] fields from RLP bytes.
60    ///
61    /// NOTE: This assumes a RLP header has already been decoded, and _just_ decodes the following
62    /// RLP fields in the following order:
63    ///
64    /// - `source_hash`
65    /// - `from`
66    /// - `kind`
67    /// - `mint`
68    /// - `value`
69    /// - `gas_limit`
70    /// - `is_system_tx`
71    /// - `input`
72    pub fn decode_inner(buf: &mut &[u8]) -> Result<Self, DecodeError> {
73        Ok(Self {
74            source_hash: Decodable::decode(buf)?,
75            from: Decodable::decode(buf)?,
76            kind: Decodable::decode(buf)?,
77            mint: Decodable::decode(buf)?,
78            value: Decodable::decode(buf)?,
79            gas_limit: Decodable::decode(buf)?,
80            is_system_tx: Decodable::decode(buf)?,
81            input: Decodable::decode(buf)?,
82        })
83    }
84
85    /// Inner encoding function that is used for both rlp [`Encodable`] trait and for calculating
86    /// hash that for eip2718 does not require rlp header
87    pub(crate) fn encode_with_signature(
88        &self,
89        signature: &Signature,
90        out: &mut dyn alloy_rlp::BufMut,
91    ) {
92        let payload_length = self.fields_len() + signature.rlp_vrs_len();
93        let header = alloy_rlp::Header { list: true, payload_length };
94        header.encode(out);
95        self.encode_fields(out);
96        signature.write_rlp_vrs(out);
97    }
98
99    /// Output the length of the RLP signed transaction encoding, _without_ a RLP string header.
100    pub fn payload_len_with_signature_without_header(&self, signature: &Signature) -> usize {
101        let payload_length = self.fields_len() + signature.rlp_vrs_len();
102        // 'transaction type byte length' + 'header length' + 'payload length'
103        1 + length_of_length(payload_length) + payload_length
104    }
105
106    /// Output the length of the RLP signed transaction encoding. This encodes with a RLP header.
107    pub fn payload_len_with_signature(&self, signature: &Signature) -> usize {
108        let len = self.payload_len_with_signature_without_header(signature);
109        length_of_length(len) + len
110    }
111
112    /// Get transaction type
113    pub(crate) const fn tx_type(&self) -> u8 {
114        DEPOSIT_TX_TYPE_ID
115    }
116
117    /// Calculates a heuristic for the in-memory size of the [DepositTransaction] transaction.
118    pub fn size(&self) -> usize {
119        mem::size_of::<B256>() + // source_hash
120        mem::size_of::<Address>() + // from
121        self.kind.size() + // to
122        mem::size_of::<U256>() + // mint
123        mem::size_of::<U256>() + // value
124        mem::size_of::<U256>() + // gas_limit
125        mem::size_of::<bool>() + // is_system_transaction
126        self.input.len() // input
127    }
128
129    /// Encodes the legacy transaction in RLP for signing.
130    pub(crate) fn encode_for_signing(&self, out: &mut dyn alloy_rlp::BufMut) {
131        out.put_u8(self.tx_type());
132        alloy_rlp::Header { list: true, payload_length: self.fields_len() }.encode(out);
133        self.encode_fields(out);
134    }
135
136    /// Outputs the length of the signature RLP encoding for the transaction.
137    pub(crate) fn payload_len_for_signature(&self) -> usize {
138        let payload_length = self.fields_len();
139        // 'transaction type byte length' + 'header length' + 'payload length'
140        1 + length_of_length(payload_length) + payload_length
141    }
142
143    fn encoded_len_with_signature(&self, signature: &Signature) -> usize {
144        // this counts the tx fields and signature fields
145        let payload_length = self.fields_len() + signature.rlp_vrs_len();
146
147        // this counts:
148        // * tx type byte
149        // * inner header length
150        // * inner payload length
151        1 + alloy_rlp::Header { list: true, payload_length }.length() + payload_length
152    }
153}

229impl SignableTransaction<Signature> for DepositTransactionRequest {

232    fn payload_len_for_signature(&self) -> usize {
233        self.payload_len_for_signature()
234    }
235
236    fn into_signed(self, signature: Signature) -> Signed<Self> {
237        let mut buf = Vec::with_capacity(self.encoded_len_with_signature(&signature));
238        self.encode_with_signature(&signature, &mut buf);
239        let hash = keccak256(&buf);
240
241        // Drop any v chain id value to ensure the signature format is correct at the time of
242        // combination for an EIP-4844 transaction. V should indicate the y-parity of the
243        // signature.
244        Signed::new_unchecked(self, signature.with_parity_bool(), hash)
245    }
246

248        self.encode_for_signing(out);

274/// An op-stack deposit transaction.
275/// See <https://github.com/ethereum-optimism/optimism/blob/develop/specs/deposits.md#the-deposited-transaction-type>
276#[derive(Clone, Debug, PartialEq, Eq, Hash, Serialize, Deserialize)]
277pub struct DepositTransaction {
278    pub nonce: u64,
279    pub source_hash: B256,
280    pub from: Address,
281    pub kind: TxKind,
282    pub mint: U256,
283    pub value: U256,
284    pub gas_limit: u128,
285    pub is_system_tx: bool,
286    pub input: Bytes,
287}

289impl DepositTransaction {
290    pub fn nonce(&self) -> &u64 {
291        &self.nonce
292    }
293
294    pub fn hash(&self) -> B256 {
295        let mut encoded = Vec::new();
296        self.encode_2718(&mut encoded);
297        B256::from_slice(alloy_primitives::keccak256(encoded).as_slice())
298    }
299
300    // /// Recovers the Ethereum address which was used to sign the transaction.
301    pub fn recover(&self) -> Result<Address, alloy_primitives::SignatureError> {
302        Ok(self.from)
303    }
304
305    pub fn chain_id(&self) -> Option<u64> {
306        None
307    }

309    pub fn encode_2718(&self, out: &mut dyn alloy_rlp::BufMut) {
310        out.put_u8(DEPOSIT_TX_TYPE_ID);
311        self.encode(out);
312    }

314    /// Encodes only the transaction's fields into the desired buffer, without a RLP header.
315    pub(crate) fn encode_fields(&self, out: &mut dyn alloy_rlp::BufMut) {
316        self.source_hash.encode(out);
317        self.from.encode(out);
318        self.kind.encode(out);
319        self.mint.encode(out);
320        self.value.encode(out);
321        self.gas_limit.encode(out);
322        self.is_system_tx.encode(out);
323        self.input.encode(out);
324    }

326    /// Calculates the length of the RLP-encoded transaction's fields.
327    pub(crate) fn fields_len(&self) -> usize {
328        let mut len = 0;
329        len += self.source_hash.length();
330        len += self.from.length();
331        len += self.kind.length();
332        len += self.mint.length();
333        len += self.value.length();
334        len += self.gas_limit.length();
335        len += self.is_system_tx.length();
336        len += self.input.length();
337        len
338    }
339
340    pub fn decode_2718(buf: &mut &[u8]) -> Result<Self, DecodeError> {
341        use bytes::Buf;
342
343        let tx_type = *buf.first().ok_or(alloy_rlp::Error::Custom("empty slice"))?;
344
345        if tx_type != DEPOSIT_TX_TYPE_ID {
346            return Err(alloy_rlp::Error::Custom("invalid tx type: expected deposit tx type"));
347        }
348
349        // Skip the tx type byte
350        buf.advance(1);
351        Self::decode(buf)
352    }
353
354    /// Decodes the inner fields from RLP bytes
355    ///
356    /// NOTE: This assumes a RLP header has already been decoded, and _just_ decodes the following
357    /// RLP fields in the following order:
358    ///
359    /// - `source_hash`
360    /// - `from`
361    /// - `kind`
362    /// - `mint`
363    /// - `value`
364    /// - `gas_limit`
365    /// - `is_system_tx`
366    /// - `input`
367    pub fn decode_inner(buf: &mut &[u8]) -> Result<Self, DecodeError> {
368        Ok(Self {
369            nonce: 0,
370            source_hash: Decodable::decode(buf)?,
371            from: Decodable::decode(buf)?,
372            kind: Decodable::decode(buf)?,
373            mint: Decodable::decode(buf)?,
374            value: Decodable::decode(buf)?,
375            gas_limit: Decodable::decode(buf)?,
376            is_system_tx: Decodable::decode(buf)?,
377            input: Decodable::decode(buf)?,
378        })
379    }
380}

382impl Encodable for DepositTransaction {
383    fn encode(&self, out: &mut dyn bytes::BufMut) {
384        RlpHeader { list: true, payload_length: self.fields_len() }.encode(out);
385        self.encode_fields(out);
386    }
387}
388
389impl Decodable for DepositTransaction {
390    fn decode(buf: &mut &[u8]) -> alloy_rlp::Result<Self> {
391        let header = RlpHeader::decode(buf)?;
392        let remaining_len = buf.len();
393        if header.payload_length > remaining_len {
394            return Err(alloy_rlp::Error::InputTooShort);
395        }
396
397        Self::decode_inner(buf)
398    }
