25/// Depending on the configured chain id and block number this should apply any specific changes
26///
27/// - checks for prevrandao mixhash after merge
28/// - applies chain specifics: on Arbitrum `block.number` is the L1 block
29///
30/// Should be called with proper chain id (retrieved from provider if not provided).
31pub fn apply_chain_and_block_specific_env_changes<N: Network>(
32    env: &mut revm::primitives::Env,
33    block: &N::BlockResponse,
34) {
35    if let Ok(chain) = NamedChain::try_from(env.cfg.chain_id) {
36        let block_number = block.header().number();
37
38        match chain {
39            NamedChain::Mainnet => {
40                // after merge difficulty is supplanted with prevrandao EIP-4399
41                if block_number >= 15_537_351u64 {
42                    env.block.difficulty = env.block.prevrandao.unwrap_or_default().into();
43                }
44
45                return;
46            }
47            NamedChain::Arbitrum |
48            NamedChain::ArbitrumGoerli |
49            NamedChain::ArbitrumNova |
50            NamedChain::ArbitrumTestnet => {
51                // on arbitrum `block.number` is the L1 block which is included in the
52                // `l1BlockNumber` field
53                if let Some(l1_block_number) = block
54                    .other_fields()
55                    .and_then(|other| other.get("l1BlockNumber").cloned())
56                    .and_then(|l1_block_number| {
57                        serde_json::from_value::<U256>(l1_block_number).ok()
58                    })
59                {
60                    env.block.number = l1_block_number;
61                }
62            }
63            _ => {}
64        }
65    }
66
67    // if difficulty is `0` we assume it's past merge
68    if block.header().difficulty().is_zero() {
69        env.block.difficulty = env.block.prevrandao.unwrap_or_default().into();
70    }
71}

73/// Given an ABI and selector, it tries to find the respective function.
74pub fn get_function<'a>(
75    contract_name: &str,
76    selector: Selector,
77    abi: &'a JsonAbi,
78) -> eyre::Result<&'a Function> {
79    abi.functions()
80        .find(|func| func.selector() == selector)
81        .ok_or_else(|| eyre::eyre!("{contract_name} does not have the selector {selector}"))
82}

84/// Configures the env for the transaction
85pub fn configure_tx_env(env: &mut revm::primitives::Env, tx: &Transaction) {
86    env.tx.caller = tx.from;
87    env.tx.gas_limit = tx.gas as u64;
88    env.tx.gas_price = U256::from(tx.gas_price.unwrap_or_default());
89    env.tx.gas_priority_fee = tx.max_priority_fee_per_gas.map(U256::from);
90    env.tx.nonce = Some(tx.nonce);
91    env.tx.access_list = tx.access_list.clone().unwrap_or_default().0.into_iter().collect();
92    env.tx.value = tx.value.to();
93    env.tx.data = alloy_primitives::Bytes(tx.input.0.clone());
94    env.tx.transact_to = tx.to.map(TxKind::Call).unwrap_or(TxKind::Create)
95}

97/// Get the gas used, accounting for refunds
98pub fn gas_used(spec: SpecId, spent: u64, refunded: u64) -> u64 {
99    let refund_quotient = if SpecId::enabled(spec, SpecId::LONDON) { 5 } else { 2 };
100    spent - (refunded).min(spent / refund_quotient)
101}

103fn get_create2_factory_call_inputs(salt: U256, inputs: CreateInputs) -> CallInputs {
104    let calldata = [&salt.to_be_bytes::<32>()[..], &inputs.init_code[..]].concat();
105    CallInputs {
106        caller: inputs.caller,
107        bytecode_address: DEFAULT_CREATE2_DEPLOYER,
108        target_address: DEFAULT_CREATE2_DEPLOYER,
109        scheme: CallScheme::Call,
110        value: CallValue::Transfer(inputs.value),
111        input: calldata.into(),
112        gas_limit: inputs.gas_limit,
113        is_static: false,
114        return_memory_offset: 0..0,
115        is_eof: false,
116    }
117}

119/// Used for routing certain CREATE2 invocations through [DEFAULT_CREATE2_DEPLOYER].
120///
121/// Overrides create hook with CALL frame if [InspectorExt::should_use_create2_factory] returns
122/// true. Keeps track of overridden frames and handles outcome in the overridden insert_call_outcome
123/// hook by inserting decoded address directly into interpreter.
124///
125/// Should be installed after [revm::inspector_handle_register] and before any other registers.
126pub fn create2_handler_register<DB: revm::Database, I: InspectorExt<DB>>(
127    handler: &mut EvmHandler<'_, I, DB>,
128) {
129    let create2_overrides = Rc::<RefCell<Vec<_>>>::new(RefCell::new(Vec::new()));
130
131    let create2_overrides_inner = create2_overrides.clone();
132    let old_handle = handler.execution.create.clone();
133    handler.execution.create =
134        Arc::new(move |ctx, mut inputs| -> Result<FrameOrResult, EVMError<DB::Error>> {
135            let CreateScheme::Create2 { salt } = inputs.scheme else {
136                return old_handle(ctx, inputs);
137            };
138            if !ctx.external.should_use_create2_factory(&mut ctx.evm, &mut inputs) {
139                return old_handle(ctx, inputs);
140            }
141
142            let gas_limit = inputs.gas_limit;
143
144            // Generate call inputs for CREATE2 factory.
145            let mut call_inputs = get_create2_factory_call_inputs(salt, *inputs);
146
147            // Call inspector to change input or return outcome.
148            let outcome = ctx.external.call(&mut ctx.evm, &mut call_inputs);
149
150            // Push data about current override to the stack.
151            create2_overrides_inner
152                .borrow_mut()
153                .push((ctx.evm.journaled_state.depth(), call_inputs.clone()));
154
155            // Sanity check that CREATE2 deployer exists.
156            let code_hash = ctx.evm.load_account(DEFAULT_CREATE2_DEPLOYER)?.info.code_hash;
157            if code_hash == KECCAK_EMPTY {
158                return Ok(FrameOrResult::Result(FrameResult::Call(CallOutcome {
159                    result: InterpreterResult {
160                        result: InstructionResult::Revert,
161                        output: "missing CREATE2 deployer".into(),
162                        gas: Gas::new(gas_limit),
163                    },
164                    memory_offset: 0..0,
165                })))
166            }
167
168            // Handle potential inspector override.
169            if let Some(outcome) = outcome {
170                return Ok(FrameOrResult::Result(FrameResult::Call(outcome)));
171            }
172
173            // Create CALL frame for CREATE2 factory invocation.
174            let mut frame_or_result = ctx.evm.make_call_frame(&call_inputs);
175
176            if let Ok(FrameOrResult::Frame(frame)) = &mut frame_or_result {
177                ctx.external
178                    .initialize_interp(&mut frame.frame_data_mut().interpreter, &mut ctx.evm)
179            }
180            frame_or_result
181        });
182
183    let create2_overrides_inner = create2_overrides;
184    let old_handle = handler.execution.insert_call_outcome.clone();
185    handler.execution.insert_call_outcome =
186        Arc::new(move |ctx, frame, shared_memory, mut outcome| {
187            // If we are on the depth of the latest override, handle the outcome.
188            if create2_overrides_inner
189                .borrow()
190                .last()
191                .map_or(false, |(depth, _)| *depth == ctx.evm.journaled_state.depth())
192            {
193                let (_, call_inputs) = create2_overrides_inner.borrow_mut().pop().unwrap();
194                outcome = ctx.external.call_end(&mut ctx.evm, &call_inputs, outcome);
195
196                // Decode address from output.
197                let address = match outcome.instruction_result() {
198                    return_ok!() => Address::try_from(outcome.output().as_ref())
199                        .map_err(|_| {
200                            outcome.result = InterpreterResult {
201                                result: InstructionResult::Revert,
202                                output: "invalid CREATE2 factory output".into(),
203                                gas: Gas::new(call_inputs.gas_limit),
204                            };
205                        })
206                        .ok(),
207                    _ => None,
208                };
209                frame
210                    .frame_data_mut()
211                    .interpreter
212                    .insert_create_outcome(CreateOutcome { address, result: outcome.result });
213
214                Ok(())
215            } else {
216                old_handle(ctx, frame, shared_memory, outcome)
217            }
218        });
219}

221/// Adds Alphanet P256 precompile to the list of loaded precompiles.
222pub fn alphanet_handler_register<DB: revm::Database, I: InspectorExt<DB>>(
223    handler: &mut EvmHandler<'_, I, DB>,
224) {
225    let prev = handler.pre_execution.load_precompiles.clone();
226    handler.pre_execution.load_precompiles = Arc::new(move || {
227        let mut loaded_precompiles = prev();
228
229        loaded_precompiles.extend([ALPHANET_P256]);
230
231        loaded_precompiles
232    });
233}

235/// Creates a new EVM with the given inspector.
236pub fn new_evm_with_inspector<'a, DB, I>(
237    db: DB,
238    env: revm::primitives::EnvWithHandlerCfg,
239    inspector: I,
240) -> revm::Evm<'a, I, DB>
241where
242    DB: revm::Database,
243    I: InspectorExt<DB>,
244{
245    let revm::primitives::EnvWithHandlerCfg { env, handler_cfg } = env;
246
247    // NOTE: We could use `revm::Evm::builder()` here, but on the current patch it has some
248    // performance issues.
249    /*
250    revm::Evm::builder()
251        .with_db(db)
252        .with_env(env)
253        .with_external_context(inspector)
254        .with_handler_cfg(handler_cfg)
255        .append_handler_register(revm::inspector_handle_register)
256        .append_handler_register(create2_handler_register)
257        .build()
258    */
259
260    let mut handler = revm::Handler::new(handler_cfg);
261    handler.append_handler_register_plain(revm::inspector_handle_register);
262    if inspector.is_alphanet() {
263        handler.append_handler_register_plain(alphanet_handler_register);
264    }
265    handler.append_handler_register_plain(create2_handler_register);
266
267    let context = revm::Context::new(revm::EvmContext::new_with_env(db, env), inspector);
268
269    revm::Evm::new(context, handler)
270}

285pub fn new_evm_with_existing_context<'a, DB, I>(
286    inner: revm::InnerEvmContext<DB>,
287    inspector: I,
288) -> revm::Evm<'a, I, DB>
289where
290    DB: revm::Database,
291    I: InspectorExt<DB>,
292{
293    let handler_cfg = HandlerCfg::new(inner.spec_id());
294
295    let mut handler = revm::Handler::new(handler_cfg);
296    handler.append_handler_register_plain(revm::inspector_handle_register);
297    if inspector.is_alphanet() {
298        handler.append_handler_register_plain(alphanet_handler_register);
299    }
300    handler.append_handler_register_plain(create2_handler_register);
301
302    let context =
303        revm::Context::new(revm::EvmContext { inner, precompiles: Default::default() }, inspector);
304    revm::Evm::new(context, handler)
305}
