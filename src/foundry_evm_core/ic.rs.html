4/// Maps from program counter to instruction counter.
5///
6/// Inverse of [`IcPcMap`].
7#[derive(Debug, Clone)]
8pub struct PcIcMap {
9    pub inner: FxHashMap<usize, usize>,
10}
11
12impl PcIcMap {
13    /// Creates a new `PcIcMap` for the given code.
14    pub fn new(code: &[u8]) -> Self {
15        Self { inner: make_map::<true>(code) }
16    }
17
18    /// Returns the length of the map.
19    pub fn len(&self) -> usize {
20        self.inner.len()
21    }
22
23    /// Returns `true` if the map is empty.
24    pub fn is_empty(&self) -> bool {
25        self.inner.is_empty()
26    }
27
28    /// Returns the instruction counter for the given program counter.
29    pub fn get(&self, pc: usize) -> Option<usize> {
30        self.inner.get(&pc).copied()
31    }
32}

34/// Map from instruction counter to program counter.
35///
36/// Inverse of [`PcIcMap`].

41impl IcPcMap {

45    }
46
47    /// Returns the length of the map.
48    pub fn len(&self) -> usize {
49        self.inner.len()
50    }
51
52    /// Returns `true` if the map is empty.
53    pub fn is_empty(&self) -> bool {
54        self.inner.is_empty()
55    }
56
57    /// Returns the program counter for the given instruction counter.
58    pub fn get(&self, ic: usize) -> Option<usize> {

60    }

63fn make_map<const PC_FIRST: bool>(code: &[u8]) -> FxHashMap<usize, usize> {
64    let mut map = FxHashMap::default();
65
66    let mut pc = 0;
67    let mut cumulative_push_size = 0;
68    while pc < code.len() {
69        let ic = pc - cumulative_push_size;
70        if PC_FIRST {
71            map.insert(pc, ic);
72        } else {
73            map.insert(ic, pc);
74        }
75
76        if (PUSH1..=PUSH32).contains(&code[pc]) {
77            // Skip the push bytes.
78            let push_size = (code[pc] - PUSH0) as usize;
79            pc += push_size;
80            cumulative_push_size += push_size;
81        }
82
83        pc += 1;
84    }
85    map
