63        identified_contracts: HashMap<Address, String>,

65        breakpoints: Breakpoints,

70    /// Starts the debugger TUI. Terminates the current process on failure or user exit.
71    pub fn run_exit(mut self) -> ! {
72        let code = match self.try_run() {
73            Ok(ExitReason::CharExit) => 0,
74            Err(e) => {
75                println!("{e}");
76                1
77            }
78        };
79        std::process::exit(code)
80    }
81
82    /// Starts the debugger TUI.
83    pub fn try_run(&mut self) -> Result<ExitReason> {
84        eyre::ensure!(!self.debug_arena.is_empty(), "debug arena is empty");
85
86        let backend = CrosstermBackend::new(io::stdout());
87        let terminal = Terminal::new(backend)?;
88        TerminalGuard::with(terminal, |terminal| self.try_run_real(terminal))
89    }
90
91    #[instrument(target = "debugger", name = "run", skip_all, ret)]
92    fn try_run_real(&mut self, terminal: &mut DebuggerTerminal) -> Result<ExitReason> {
93        // Create the context.
94        let mut cx = DebuggerContext::new(self);
95
96        cx.init();
97
98        // Create an event listener in a different thread.
99        let (tx, rx) = mpsc::channel();
100        thread::Builder::new()
101            .name("event-listener".into())
102            .spawn(move || Self::event_listener(tx))
103            .expect("failed to spawn thread");
104
105        // Start the event loop.
106        loop {
107            cx.draw(terminal)?;
108            match cx.handle_event(rx.recv()?) {
109                ControlFlow::Continue(()) => {}
110                ControlFlow::Break(reason) => return Ok(reason),
111            }
112        }
113    }
114
115    fn event_listener(tx: mpsc::Sender<Event>) {
116        // This is the recommend tick rate from `ratatui`, based on their examples
117        let tick_rate = Duration::from_millis(200);
118
119        let mut last_tick = Instant::now();
120        loop {
121            // Poll events since last tick - if last tick is greater than tick_rate, we
122            // demand immediate availability of the event. This may affect interactivity,
123            // but I'm not sure as it is hard to test.
124            if event::poll(tick_rate.saturating_sub(last_tick.elapsed())).unwrap() {
125                let event = event::read().unwrap();
126                if tx.send(event).is_err() {
127                    return;
128                }
129            }
130
131            // Force update if time has passed
132            if last_tick.elapsed() > tick_rate {
133                last_tick = Instant::now();
134            }
135        }
136    }
137}

141type PanicHandler = Box<dyn Fn(&std::panic::PanicInfo<'_>) + 'static + Sync + Send>;

143/// Handles terminal state.
144#[must_use]
145struct TerminalGuard<B: Backend + io::Write> {
146    terminal: Terminal<B>,
147    hook: Option<Arc<PanicHandler>>,
148}
149
150impl<B: Backend + io::Write> TerminalGuard<B> {
151    fn with<T>(terminal: Terminal<B>, mut f: impl FnMut(&mut Terminal<B>) -> T) -> T {
152        let mut guard = Self { terminal, hook: None };
153        guard.setup();
154        f(&mut guard.terminal)
155    }
156
157    fn setup(&mut self) {
158        let previous = Arc::new(std::panic::take_hook());
159        self.hook = Some(previous.clone());
160        // We need to restore the terminal state before displaying the panic message.
161        // TODO: Use `std::panic::update_hook` when it's stable
162        std::panic::set_hook(Box::new(move |info| {
163            Self::half_restore(&mut std::io::stdout());
164            (previous)(info)
165        }));
166
167        let _ = enable_raw_mode();
168        let _ = execute!(*self.terminal.backend_mut(), EnterAlternateScreen, EnableMouseCapture);
169        let _ = self.terminal.hide_cursor();
170        let _ = self.terminal.clear();
171    }
172
173    fn restore(&mut self) {
174        if !std::thread::panicking() {
175            // Drop the current hook to guarantee that `self.hook` is the only reference to it.
176            let _ = std::panic::take_hook();
177            // Restore the previous panic hook.
178            let prev = self.hook.take().unwrap();
179            let prev = match Arc::try_unwrap(prev) {
180                Ok(prev) => prev,
181                Err(_) => unreachable!("`self.hook` is not the only reference to the panic hook"),
182            };
183            std::panic::set_hook(prev);
184
185            // NOTE: Our panic handler calls this function, so we only have to call it here if we're
186            // not panicking.
187            Self::half_restore(self.terminal.backend_mut());
188        }
189
190        let _ = self.terminal.show_cursor();
191    }
192
193    fn half_restore(w: &mut impl io::Write) {
194        let _ = disable_raw_mode();
195        let _ = execute!(*w, LeaveAlternateScreen, DisableMouseCapture);
196    }
197}
198
199impl<B: Backend + io::Write> Drop for TerminalGuard<B> {
200    #[inline]
201    fn drop(&mut self) {
202        self.restore();
203    }
