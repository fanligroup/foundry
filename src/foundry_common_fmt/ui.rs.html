12/// length of the name column for pretty formatting `{:>20}{value}`
13const NAME_COLUMN_LEN: usize = 20usize;

15/// Helper trait to format Ethereum types.
16///
17/// # Examples
18///
19/// ```
20/// use foundry_common_fmt::UIfmt;
21///
22/// let boolean: bool = true;
23/// let string = boolean.pretty();
24/// ```
25pub trait UIfmt {
26    /// Return a prettified string version of the value
27    fn pretty(&self) -> String;
28}
29
30impl<T: UIfmt> UIfmt for &T {
31    fn pretty(&self) -> String {
32        (*self).pretty()
33    }
34}

36impl<T: UIfmt> UIfmt for Option<T> {
37    fn pretty(&self) -> String {
38        if let Some(ref inner) = self {
39            inner.pretty()
40        } else {
41            String::new()
42        }
43    }
44}

46impl<T: UIfmt> UIfmt for [T] {
47    fn pretty(&self) -> String {
48        if !self.is_empty() {
49            let mut s = String::with_capacity(self.len() * 64);
50            s.push_str("[\n");
51            for item in self {
52                for line in item.pretty().lines() {
53                    s.push('\t');
54                    s.push_str(line);
55                    s.push('\n');
56                }
57            }
58            s.push(']');
59            s
60        } else {
61            "[]".to_string()
62        }
63    }
64}

159impl UIfmt for AnyTransactionReceipt {
160    fn pretty(&self) -> String {
161        let Self {
162            inner:
163                TransactionReceipt {
164                    transaction_hash,
165                    transaction_index,
166                    block_hash,
167                    block_number,
168                    from,
169                    to,
170                    gas_used,
171                    contract_address,
172                    state_root,
173                    effective_gas_price,
174                    inner:
175                        AnyReceiptEnvelope {
176                            r#type: transaction_type,
177                            inner:
178                                ReceiptWithBloom {
179                                    receipt: Receipt { status, cumulative_gas_used, logs },
180                                    logs_bloom,
181                                },
182                        },
183                    blob_gas_price,
184                    blob_gas_used,
185                    authorization_list,
186                },
187            other,
188        } = self;
189
190        let mut pretty = format!(
191            "
192blockHash               {}
193blockNumber             {}
194contractAddress         {}
195cumulativeGasUsed       {}
196effectiveGasPrice       {}
197from                    {}
198gasUsed                 {}
199logs                    {}
200logsBloom               {}
201root                    {}
202status                  {}
203transactionHash         {}
204transactionIndex        {}
205type                    {}
206blobGasPrice            {}
207blobGasUsed             {}
208authorizationList       {}",
209            block_hash.pretty(),
210            block_number.pretty(),
211            contract_address.pretty(),
212            cumulative_gas_used.pretty(),
213            effective_gas_price.pretty(),
214            from.pretty(),
215            gas_used.pretty(),
216            serde_json::to_string(&logs).unwrap(),
217            logs_bloom.pretty(),
218            state_root.pretty(),
219            status.pretty(),
220            transaction_hash.pretty(),
221            transaction_index.pretty(),
222            transaction_type,
223            blob_gas_price.pretty(),
224            blob_gas_used.pretty(),
225            authorization_list
226                .as_ref()
227                .map(|l| serde_json::to_string(&l).unwrap())
228                .unwrap_or_default(),
229        );
230
231        if let Some(to) = to {
232            pretty.push_str(&format!("\nto                      {}", to.pretty()));
233        }
234
235        // additional captured fields
236        for (key, val) in other.iter() {
237            pretty.push_str(&format!("\n{key}             {val}"));
238        }
239
240        pretty
241    }
242}

292impl UIfmt for OtherFields {
293    fn pretty(&self) -> String {
294        let mut s = String::with_capacity(self.len() * 30);
295        if !self.is_empty() {
296            s.push('\n');
297        }
298        for (key, value) in self.iter() {
299            let val = EthValue::from(value.clone()).pretty();
300            let offset = NAME_COLUMN_LEN.saturating_sub(key.len());
301            s.push_str(key);
302            s.extend(std::iter::repeat(' ').take(offset + 1));
303            s.push_str(&val);
304            s.push('\n');
305        }
306        s
307    }
308}

527/// Various numerical ethereum types used for pretty printing
528#[derive(Clone, Debug, Deserialize)]
529#[serde(untagged)]
530#[allow(missing_docs)]
531pub enum EthValue {
532    U64(U64),
533    U256(U256),
534    U64Array(Vec<U64>),
535    U256Array(Vec<U256>),
536    Other(serde_json::Value),
537}
538
539impl From<serde_json::Value> for EthValue {
540    fn from(val: serde_json::Value) -> Self {
541        serde_json::from_value(val).expect("infallible")
542    }
543}
544
545impl UIfmt for EthValue {
546    fn pretty(&self) -> String {
547        match self {
548            Self::U64(num) => num.pretty(),
549            Self::U256(num) => num.pretty(),
550            Self::U64Array(arr) => arr.pretty(),
551            Self::U256Array(arr) => arr.pretty(),
552            Self::Other(val) => val.to_string().trim_matches('"').to_string(),
553        }
554    }
555}

557/// Returns the `UiFmt::pretty()` formatted attribute of the transactions
558pub fn get_pretty_tx_attr(transaction: &Transaction, attr: &str) -> Option<String> {
559    match attr {
560        "blockHash" | "block_hash" => Some(transaction.block_hash.pretty()),
561        "blockNumber" | "block_number" => Some(transaction.block_number.pretty()),
562        "from" => Some(transaction.from.pretty()),
563        "gas" => Some(transaction.gas.pretty()),
564        "gasPrice" | "gas_price" => Some(transaction.gas_price.pretty()),
565        "hash" => Some(transaction.hash.pretty()),
566        "input" => Some(transaction.input.pretty()),
567        "nonce" => Some(transaction.nonce.to_string()),
568        "s" => transaction.signature.map(|s| B256::from(s.s).pretty()),
569        "r" => transaction.signature.map(|s| B256::from(s.r).pretty()),
570        "to" => Some(transaction.to.pretty()),
571        "transactionIndex" | "transaction_index" => Some(transaction.transaction_index.pretty()),
572        "v" => transaction.signature.map(|s| s.v.pretty()),
573        "value" => Some(transaction.value.pretty()),
574        _ => None,
575    }
576}

578/// Returns the `UiFmt::pretty()` formatted attribute of the given block
579pub fn get_pretty_block_attr(block: &AnyNetworkBlock, attr: &str) -> Option<String> {
580    match attr {
581        "baseFeePerGas" | "base_fee_per_gas" => Some(block.header.base_fee_per_gas.pretty()),
582        "difficulty" => Some(block.header.difficulty.pretty()),
583        "extraData" | "extra_data" => Some(block.header.extra_data.pretty()),
584        "gasLimit" | "gas_limit" => Some(block.header.gas_limit.pretty()),
585        "gasUsed" | "gas_used" => Some(block.header.gas_used.pretty()),
586        "hash" => Some(block.header.hash.pretty()),
587        "logsBloom" | "logs_bloom" => Some(block.header.logs_bloom.pretty()),
588        "miner" | "author" => Some(block.header.miner.pretty()),
589        "mixHash" | "mix_hash" => Some(block.header.mix_hash.pretty()),
590        "nonce" => Some(block.header.nonce.pretty()),
591        "number" => Some(block.header.number.pretty()),
592        "parentHash" | "parent_hash" => Some(block.header.parent_hash.pretty()),
593        "transactionsRoot" | "transactions_root" => Some(block.header.transactions_root.pretty()),
594        "receiptsRoot" | "receipts_root" => Some(block.header.receipts_root.pretty()),
595        "sha3Uncles" | "sha_3_uncles" => Some(block.header.uncles_hash.pretty()),
596        "size" => Some(block.size.pretty()),
597        "stateRoot" | "state_root" => Some(block.header.state_root.pretty()),
598        "timestamp" => Some(block.header.timestamp.pretty()),
599        "totalDifficulty" | "total_difficult" => Some(block.header.total_difficulty.pretty()),
600        other => {
601            if let Some(value) = block.other.get(other) {
602                let val = EthValue::from(value.clone());
603                return Some(val.pretty())
604            }
605            None
606        }
607    }
608}
