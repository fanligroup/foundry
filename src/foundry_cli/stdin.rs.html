10/// Prints a message to [`stdout`][io::stdout] and [reads a line from stdin into a String](read).
11///
12/// Returns `Result<T>`, so sometimes `T` must be explicitly specified, like in `str::parse`.
13///
14/// # Examples
15///
16/// ```no_run
17/// # use foundry_cli::prompt;
18/// let response: String = prompt!("Would you like to continue? [y/N] ")?;
19/// if !matches!(response.as_str(), "y" | "Y") {
20///     return Ok(())
21/// }
22/// # Ok::<(), Box<dyn std::error::Error>>(())
23/// ```
24#[macro_export]
25macro_rules! prompt {
26    () => {
27        $crate::stdin::parse_line()
28    };
29
30    ($($tt:tt)+) => {
31        {
32            ::std::print!($($tt)+);
33            match ::std::io::Write::flush(&mut ::std::io::stdout()) {
34                ::core::result::Result::Ok(_) => $crate::prompt!(),
35                ::core::result::Result::Err(e) => ::core::result::Result::Err(::eyre::eyre!("Could not flush stdout: {}", e))
36            }
37        }
38    };
39}

41/// Unwraps the given `Option<T>` or [reads stdin into a String](read) and parses it as `T`.
42pub fn unwrap<T>(value: Option<T>, read_line: bool) -> Result<T>
43where
44    T: FromStr,
45    T::Err: StdError + Send + Sync + 'static,
46{
47    match value {
48        Some(value) => Ok(value),
49        None => parse(read_line),
50    }
51}
52
53#[inline]
54pub fn unwrap2<A, B>(a: Option<A>, b: Option<B>) -> Result<(A, B)>
55where
56    A: FromStr,
57    B: FromStr,
58    A::Err: StdError + Send + Sync + 'static,
59    B::Err: StdError + Send + Sync + 'static,
60{
61    match (a, b) {
62        (Some(a), Some(b)) => Ok((a, b)),
63        (a, b) => Ok((unwrap(a, true)?, unwrap(b, true)?)),
64    }
65}

67/// [Reads stdin into a String](read) and parses it as `Vec<T>` using whitespaces as delimiters if
68/// the given `Vec<T>` is empty.
69pub fn unwrap_vec<T>(mut value: Vec<T>) -> Result<Vec<T>>
70where
71    T: FromStr,
72    T::Err: StdError + Send + Sync + 'static,
73{
74    if value.is_empty() {
75        let s = read(false)?;
76        value = s.split_whitespace().map(FromStr::from_str).collect::<Result<Vec<T>, _>>()?;
77    }
78
79    Ok(value)
80}

92/// Reads bytes from [`stdin`][io::stdin] into a String.
93///
94/// If `read_line` is true, stop at the first newline (the `0xA` byte).
95#[inline]
96pub fn parse<T>(read_line: bool) -> Result<T>
97where
98    T: FromStr,
99    T::Err: StdError + Send + Sync + 'static,
100{
101    read(read_line).and_then(|s| s.parse().map_err(Into::into))
102}
103
104/// Short-hand for `parse(true)`.
105#[inline]
106pub fn parse_line<T>() -> Result<T>
107where
108    T: FromStr,
109    T::Err: StdError + Send + Sync + 'static,
110{
111    parse(true)
112}

114/// Reads bytes from [`stdin`][io::stdin] into a String.
115///
116/// If `read_line` is true, stop at the first newline (the `0xA` byte).
117#[inline]
118pub fn read(read_line: bool) -> Result<String> {
119    let bytes = read_bytes(read_line)?;
120
121    if read_line {
122        // SAFETY: [BufRead::read_line] appends into a String
123        Ok(unsafe { String::from_utf8_unchecked(bytes) })
124    } else {
125        String::from_utf8(bytes).map_err(Into::into)
126    }
127}

129/// Reads bytes from [`stdin`][io::stdin].
130///
131/// If `read_line` is true, read up to the first newline excluded (the `0xA` byte).
132pub fn read_bytes(read_line: bool) -> Result<Vec<u8>> {
133    let mut stdin = io::stdin().lock();
134
135    if read_line {
136        let mut buf = String::new();
137        stdin.read_line(&mut buf)?;
138        // remove the trailing newline
139        if let Some(b'\n') = buf.as_bytes().last() {
140            buf.pop();
141        }
142        Ok(buf.into_bytes())
143    } else {
144        let mut buf = Vec::new();
145        stdin.read_to_end(&mut buf)?;
146        Ok(buf)
147    }
