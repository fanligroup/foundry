31/// Given a `Project`'s output, removes the matching ABI, Bytecode and
32/// Runtime Bytecode of the given contract.
33#[track_caller]
34pub fn remove_contract(
35    output: &mut ProjectCompileOutput,
36    path: &Path,
37    name: &str,
38) -> Result<(JsonAbi, CompactBytecode, CompactDeployedBytecode)> {
39    let contract = if let Some(contract) = output.remove(path, name) {
40        contract
41    } else {
42        let mut err = format!("could not find artifact: `{name}`");
43        if let Some(suggestion) =
44            super::did_you_mean(name, output.artifacts().map(|(name, _)| name)).pop()
45        {
46            if suggestion != name {
47                err = format!(
48                    r#"{err}
49
50        Did you mean `{suggestion}`?"#
51                );
52            }
53        }
54        eyre::bail!(err)
55    };
56
57    let abi = contract
58        .get_abi()
59        .ok_or_else(|| eyre::eyre!("contract {} does not contain abi", name))?
60        .into_owned();
61
62    let bin = contract
63        .get_bytecode()
64        .ok_or_else(|| eyre::eyre!("contract {} does not contain bytecode", name))?
65        .into_owned();
66
67    let runtime = contract
68        .get_deployed_bytecode()
69        .ok_or_else(|| eyre::eyre!("contract {} does not contain deployed bytecode", name))?
70        .into_owned();
71
72    Ok((abi, bin, runtime))
73}

75/// Helper function for finding a contract by ContractName
76// TODO: Is there a better / more ergonomic way to get the artifacts given a project and a
77// contract name?
78pub fn get_cached_entry_by_name(
79    cache: &CompilerCache<Settings>,
80    name: &str,
81) -> Result<(PathBuf, CacheEntry)> {
82    let mut cached_entry = None;
83    let mut alternatives = Vec::new();
84
85    for (abs_path, entry) in cache.files.iter() {
86        for (artifact_name, _) in entry.artifacts.iter() {
87            if artifact_name == name {
88                if cached_entry.is_some() {
89                    eyre::bail!(
90                        "contract with duplicate name `{}`. please pass the path instead",
91                        name
92                    )
93                }
94                cached_entry = Some((abs_path.to_owned(), entry.to_owned()));
95            } else {
96                alternatives.push(artifact_name);
97            }
98        }
99    }
100
101    if let Some(entry) = cached_entry {
102        return Ok(entry);
103    }
104
105    let mut err = format!("could not find artifact: `{name}`");
106    if let Some(suggestion) = super::did_you_mean(name, &alternatives).pop() {
107        err = format!(
108            r#"{err}
109
110        Did you mean `{suggestion}`?"#
111        );
112    }
113    eyre::bail!(err)
114}

116/// Returns error if constructor has arguments.
117pub fn ensure_clean_constructor(abi: &JsonAbi) -> Result<()> {
118    if let Some(constructor) = &abi.constructor {
119        if !constructor.inputs.is_empty() {
120            eyre::bail!("Contract constructor should have no arguments. Add those arguments to  `run(...)` instead, and call it with `--sig run(...)`.");
121        }
122    }
123    Ok(())
124}

126pub fn needs_setup(abi: &JsonAbi) -> bool {
127    let setup_fns: Vec<_> = abi.functions().filter(|func| func.name.is_setup()).collect();
128
129    for setup_fn in setup_fns.iter() {
130        if setup_fn.name != "setUp" {
131            println!(
132                "{} Found invalid setup function \"{}\" did you mean \"setUp()\"?",
133                "Warning:".yellow().bold(),
134                setup_fn.signature()
135            );
136        }
137    }
138
139    setup_fns.len() == 1 && setup_fns[0].name == "setUp"
140}

202/// Helpers for loading configuration.
203///
204/// This is usually implicitly implemented on a "&CmdArgs" struct via impl macros defined in
205/// `forge_config` (see [`foundry_config::impl_figment_convert`] for more details) and the impl
206/// definition on `T: Into<Config> + Into<Figment>` below.
207///
208/// Each function also has an `emit_warnings` form which does the same thing as its counterpart but
209/// also prints `Config::__warnings` to stderr
210pub trait LoadConfig {
211    /// Load and sanitize the [`Config`] based on the options provided in self
212    ///
213    /// Returns an error if loading the config failed
214    fn try_load_config(self) -> Result<Config, ExtractConfigError>;
215    /// Load and sanitize the [`Config`] based on the options provided in self
216    fn load_config(self) -> Config;
217    /// Load and sanitize the [`Config`], as well as extract [`EvmOpts`] from self
218    fn load_config_and_evm_opts(self) -> Result<(Config, EvmOpts)>;
219    /// Load [`Config`] but do not sanitize. See [`Config::sanitized`] for more information
220    fn load_config_unsanitized(self) -> Config;
221    /// Load [`Config`] but do not sanitize. See [`Config::sanitized`] for more information.
222    ///
223    /// Returns an error if loading failed
224    fn try_load_config_unsanitized(self) -> Result<Config, ExtractConfigError>;
225    /// Same as [`LoadConfig::load_config`] but also emits warnings generated
226    fn load_config_emit_warnings(self) -> Config;
227    /// Same as [`LoadConfig::load_config`] but also emits warnings generated
228    ///
229    /// Returns an error if loading failed
230    fn try_load_config_emit_warnings(self) -> Result<Config, ExtractConfigError>;
231    /// Same as [`LoadConfig::load_config_and_evm_opts`] but also emits warnings generated
232    fn load_config_and_evm_opts_emit_warnings(self) -> Result<(Config, EvmOpts)>;
233    /// Same as [`LoadConfig::load_config_unsanitized`] but also emits warnings generated
234    fn load_config_unsanitized_emit_warnings(self) -> Config;
235    fn try_load_config_unsanitized_emit_warnings(self) -> Result<Config, ExtractConfigError>;
236}

238impl<T> LoadConfig for T
239where
240    T: Into<Config> + Into<Figment>,
241{
242    fn try_load_config(self) -> Result<Config, ExtractConfigError> {
243        let figment: Figment = self.into();
244        Ok(Config::try_from(figment)?.sanitized())
245    }
246
247    fn load_config(self) -> Config {
248        self.into()
249    }
250
251    fn load_config_and_evm_opts(self) -> Result<(Config, EvmOpts)> {
252        let figment: Figment = self.into();
253
254        let mut evm_opts = figment.extract::<EvmOpts>().map_err(ExtractConfigError::new)?;
255        let config = Config::try_from(figment)?.sanitized();
256
257        // update the fork url if it was an alias
258        if let Some(fork_url) = config.get_rpc_url() {
259            trace!(target: "forge::config", ?fork_url, "Update EvmOpts fork url");
260            evm_opts.fork_url = Some(fork_url?.into_owned());
261        }
262
263        Ok((config, evm_opts))
264    }
265
266    fn load_config_unsanitized(self) -> Config {
267        let figment: Figment = self.into();
268        Config::from_provider(figment)
269    }
270
271    fn try_load_config_unsanitized(self) -> Result<Config, ExtractConfigError> {
272        let figment: Figment = self.into();
273        Config::try_from(figment)
274    }
275
276    fn load_config_emit_warnings(self) -> Config {
277        let config = self.load_config();
278        config.warnings.iter().for_each(|w| cli_warn!("{w}"));
279        config
280    }
281
282    fn try_load_config_emit_warnings(self) -> Result<Config, ExtractConfigError> {
283        let config = self.try_load_config()?;
284        config.warnings.iter().for_each(|w| cli_warn!("{w}"));
285        Ok(config)
286    }
287
288    fn load_config_and_evm_opts_emit_warnings(self) -> Result<(Config, EvmOpts)> {
289        let (config, evm_opts) = self.load_config_and_evm_opts()?;
290        config.warnings.iter().for_each(|w| cli_warn!("{w}"));
291        Ok((config, evm_opts))
292    }
293
294    fn load_config_unsanitized_emit_warnings(self) -> Config {
295        let config = self.load_config_unsanitized();
296        config.warnings.iter().for_each(|w| cli_warn!("{w}"));
297        config
298    }
299
300    fn try_load_config_unsanitized_emit_warnings(self) -> Result<Config, ExtractConfigError> {
301        let config = self.try_load_config_unsanitized()?;
302        config.warnings.iter().for_each(|w| cli_warn!("{w}"));
303        Ok(config)
304    }
305}

307/// Read contract constructor arguments from the given file.
308pub fn read_constructor_args_file(constructor_args_path: PathBuf) -> Result<Vec<String>> {
309    if !constructor_args_path.exists() {
310        eyre::bail!("Constructor args file \"{}\" not found", constructor_args_path.display());
311    }
312    let args = if constructor_args_path.extension() == Some(std::ffi::OsStr::new("json")) {
313        read_json_file(&constructor_args_path).wrap_err(format!(
314            "Constructor args file \"{}\" must encode a json array",
315            constructor_args_path.display(),
316        ))?
317    } else {
318        fs::read_to_string(constructor_args_path)?.split_whitespace().map(str::to_string).collect()
319    };
320    Ok(args)
321}

323/// A slimmed down return from the executor used for returning minimal trace + gas metering info
324#[derive(Debug)]
325pub struct TraceResult {
326    pub success: bool,
327    pub traces: Option<Traces>,
328    pub gas_used: u64,
329}
330
331impl TraceResult {
332    /// Create a new [`TraceResult`] from a [`RawCallResult`].
333    pub fn from_raw(raw: RawCallResult, trace_kind: TraceKind) -> Self {
334        let RawCallResult { gas_used, traces, reverted, .. } = raw;
335        Self { success: !reverted, traces: traces.map(|arena| vec![(trace_kind, arena)]), gas_used }
336    }
337}
338
339impl From<DeployResult> for TraceResult {
340    fn from(result: DeployResult) -> Self {
341        Self::from_raw(result.raw, TraceKind::Deployment)
342    }
343}
344
345impl TryFrom<Result<DeployResult, EvmError>> for TraceResult {
346    type Error = EvmError;
347
348    fn try_from(value: Result<DeployResult, EvmError>) -> Result<Self, Self::Error> {
349        match value {
350            Ok(result) => Ok(Self::from(result)),
351            Err(EvmError::Execution(err)) => Ok(Self::from_raw(err.raw, TraceKind::Deployment)),
352            Err(err) => Err(err),
353        }
354    }
355}

357/// labels the traces, conditionally prints them or opens the debugger
358pub async fn handle_traces(
359    mut result: TraceResult,
360    config: &Config,
361    chain: Option<Chain>,
362    labels: Vec<String>,
363    debug: bool,
364    decode_internal: bool,
365    verbose: bool,
366) -> Result<()> {
367    let labels = labels.iter().filter_map(|label_str| {
368        let mut iter = label_str.split(':');
369
370        if let Some(addr) = iter.next() {
371            if let (Ok(address), Some(label)) = (Address::from_str(addr), iter.next()) {
372                return Some((address, label.to_string()));
373            }
374        }
375        None
376    });
377    let config_labels = config.labels.clone().into_iter();
378    let mut decoder = CallTraceDecoderBuilder::new()
379        .with_labels(labels.chain(config_labels))
380        .with_signature_identifier(SignaturesIdentifier::new(
381            Config::foundry_cache_dir(),
382            config.offline,
383        )?)
384        .build();
385
386    let mut etherscan_identifier = EtherscanIdentifier::new(config, chain)?;
387    if let Some(etherscan_identifier) = &mut etherscan_identifier {
388        for (_, trace) in result.traces.as_deref_mut().unwrap_or_default() {
389            decoder.identify(trace, etherscan_identifier);
390        }
391    }
392
393    if decode_internal {
394        let sources = if let Some(etherscan_identifier) = &etherscan_identifier {
395            etherscan_identifier.get_compiled_contracts().await?
396        } else {
397            Default::default()
398        };
399        decoder.debug_identifier = Some(DebugTraceIdentifier::new(sources));
400    }
401
402    if debug {
403        let sources = if let Some(etherscan_identifier) = etherscan_identifier {
404            etherscan_identifier.get_compiled_contracts().await?
405        } else {
406            Default::default()
407        };
408        let mut debugger = Debugger::builder()
409            .traces(result.traces.expect("missing traces"))
410            .decoder(&decoder)
411            .sources(sources)
412            .build();
413        debugger.try_run()?;
414    } else {
415        print_traces(&mut result, &decoder, verbose).await?;
416    }
417
418    Ok(())
419}

421pub async fn print_traces(
422    result: &mut TraceResult,
423    decoder: &CallTraceDecoder,
424    verbose: bool,
425) -> Result<()> {
426    let traces = result.traces.as_mut().expect("No traces found");
427
428    println!("Traces:");
429    for (_, arena) in traces {
430        decode_trace_arena(arena, decoder).await?;
431        println!("{}", render_trace_arena_with_bytecodes(arena, verbose));
432    }
433    println!();
434
435    if result.success {
436        println!("{}", "Transaction successfully executed.".green());
437    } else {
438        println!("{}", "Transaction failed.".red());
439    }
440
441    println!("Gas used: {}", result.gas_used);
442    Ok(())
