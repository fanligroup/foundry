1use crate::utils::parse_ether_value;
2use alloy_primitives::{U256, U64};
3use clap::Parser;
4use serde::Serialize;
5
6#[derive(Clone, Debug, Serialize, Parser)]
7#[command(next_help_heading = "Transaction options")]
8pub struct TransactionOpts {
9    /// Gas limit for the transaction.
10    #[arg(long, env = "ETH_GAS_LIMIT")]
11    pub gas_limit: Option<U256>,
12
13    /// Gas price for legacy transactions, or max fee per gas for EIP1559 transactions, either
14    /// specified in wei, or as a string with a unit type.
15    ///
16    /// Examples: 1ether, 10gwei, 0.01ether
17    #[arg(
18        long,
19        env = "ETH_GAS_PRICE",
20        value_parser = parse_ether_value,
21        value_name = "PRICE"
22    )]
23    pub gas_price: Option<U256>,
24
25    /// Max priority fee per gas for EIP1559 transactions.
26    #[arg(
27        long,
28        env = "ETH_PRIORITY_GAS_PRICE",
29        value_parser = parse_ether_value,
30        value_name = "PRICE"
31    )]
32    pub priority_gas_price: Option<U256>,
33
34    /// Ether to send in the transaction, either specified in wei, or as a string with a unit type.
35    ///
36    ///
37    ///
38    /// Examples: 1ether, 10gwei, 0.01ether
39    #[arg(long, value_parser = parse_ether_value)]
40    pub value: Option<U256>,
41
42    /// Nonce for the transaction.
43    #[arg(long)]
44    pub nonce: Option<U64>,
45
46    /// Send a legacy transaction instead of an EIP1559 transaction.
47    ///
48    /// This is automatically enabled for common networks without EIP1559.
49    #[arg(long)]
50    pub legacy: bool,
51
52    /// Send a EIP-4844 blob transaction.
53    #[arg(long, conflicts_with = "legacy")]
54    pub blob: bool,
55
56    /// Gas price for EIP-4844 blob transaction.
57    #[arg(long, conflicts_with = "legacy", value_parser = parse_ether_value, env = "ETH_BLOB_GAS_PRICE", value_name = "BLOB_PRICE")]
58    pub blob_gas_price: Option<U256>,
59
60    /// EIP-7702 authorization list.
61    ///
62    /// Can be either a hex-encoded signed authorization or an address.
63    #[arg(long, conflicts_with_all = &["legacy", "blob"])]
64    pub auth: Option<String>,
65
66    /// EIP-2930 access list.
67    ///
68    /// Accepts either a JSON-encoded access list or an empty value to create the access list
69    /// via an RPC call to `eth_createAccessList`. To retrieve only the access list portion, use
70    /// the `cast access-list` command.
71    #[arg(long)]
72    pub access_list: Option<Option<String>>,
73}
